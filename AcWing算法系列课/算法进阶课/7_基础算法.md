# 7 基础算法

## Manacher算法

模板：

```c++
int n;
char A[N], S[N * 2];
int P[N * 2];

void init()
{
    n = strlen(A);
    S[0] = '$', S[1] = '#';
    int j = 2;
    for (int i = 0; i < n; i++)
    {
        S[j++] = A[i];
        S[j++] = '#';
    }
    S[j++] = '^';
    n = j;
}

void manacher()
{
    int mid = 0, r = 0;
    for (int i = 1; i < n; i++)
    {
        int j = mid * 2 - i;
        if (i <= r) P[i] = std::min(P[j], r - i + 1);
        else P[i] = 1;
        while (S[i + P[i]] == S[i - P[i]]) ++P[i];
        if (i + P[i] - 1 > r)
        {
            mid = i;
            r = i + P[i] - 1;
        }
    }
}
```



### manacher算法

题目：https://www.acwing.com/problem/content/description/3190/

题解：https://www.acwing.com/solution/content/66912/

```c++
#include <cstdio>
#include <cstring>
#include <iostream>
#include <algorithm>

using std::cin, std::cout;
using LL = long long;

const int N = 1e7 + 5;

int n;
char A[N], S[N * 2];
int P[N * 2];

void init()
{
    n = strlen(A);
    S[0] = '$', S[1] = '#';
    int j = 2;
    for (int i = 0; i < n; i++)
    {
        S[j++] = A[i];
        S[j++] = '#';
    }
    S[j++] = '^';
    n = j;
}

void manacher()
{
    int mid = 0, r = 0;
    for (int i = 1; i < n; i++)
    {
        int j = mid * 2 - i;
        if (i <= r) P[i] = std::min(P[j], r - i + 1);
        else P[i] = 1;
        while (S[i + P[i]] == S[i - P[i]]) ++P[i];
        if (i + P[i] - 1 > r)
        {
            mid = i;
            r = i + P[i] - 1;
        }
    }
}

int main()
{
    std::ios::sync_with_stdio(false);
    cin.tie(nullptr), cout.tie(nullptr);

    cin >> A;
    init();

    manacher();

    int res = 0;
    for (int i = 1; i < n; i++) res = std::max(res, P[i] - 1);
    cout << res << '\n';

    return 0;
}
```
