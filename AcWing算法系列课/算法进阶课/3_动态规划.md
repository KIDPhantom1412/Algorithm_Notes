# 动态规划

## 基环树DP

### 岛屿

题目：https://www.acwing.com/problem/content/description/360/

题解：https://www.acwing.com/solution/content/166480/

```c++
#include <cstdio>
#include <iostream>
#include <algorithm>
#include <numeric>
#include <vector>

using LL = long long;
using std::cin, std::cout;
using std::vector;

const int N = int(1e6) + 5, M = N * 2;

int n;
int H[N];
struct
{
    int v, w, n;
} E[M];
int idx = 1;

void add(int a, int b, int w)
{
    E[++idx] = { b, w, H[a] }, H[a] = idx;
}

bool st[N];
vector<vector<int>> circles;
int fw[N];
int stk[N], top;
bool inStk[N];
LL sv[N];

void findCircle(int u, int from)
{
    inStk[u] = st[u] = true;
    stk[++top] = u;

    for (int i = H[u]; i; i = E[i].n)
        if (i != (from ^ 1))
        {
            int v = E[i].v, w = E[i].w;
            fw[v] = w;
            if (!st[v])
            {
                findCircle(v, i);
            }
            else if (inStk[v])
            {
                vector<int> circle;
                LL sum = w;
                for (int j = top; stk[j] != v; j--)
                {
                    circle.push_back(stk[j]);
                    sv[stk[j]] = sum;
                    sum += fw[stk[j]];
                }
                circle.push_back(v);
                sv[v] = sum;
                circles.push_back(circle);
            }
        }

    inStk[u] = false;
    top--;
}

LL getHeight(int u, LL& mx)
{
    st[u] = true;

    LL h1 = 0, h2 = 0;
    for (int i = H[u]; i; i = E[i].n)
    {
        int v = E[i].v, w = E[i].w;
        if (!st[v])
        {
            LL h = getHeight(v, mx) + w;
            if (h > h1) h2 = h1, h1 = h;
            else if (h > h2) h2 = h;
        }
    }

    mx = std::max(mx, h1 + h2);
    return h1;
}

int main()
{
    std::ios::sync_with_stdio(false);
    cin.tie(nullptr), cout.tie(nullptr);

    cin >> n;
    for (int i = 1; i <= n; i++)
    {
        int a, l; cin >> a >> l;
        add(i, a, l), add(a, i, l);
    }

    for (int i = 1; i <= n; i++)
        if (!st[i])
            findCircle(i, -1);

    std::fill_n(st, N, false);
    for (auto& circle : circles)
        for (int u : circle)
        {
            st[u] = true;
        }

    LL res = 0;
    for (auto& circle : circles)
    {
        LL mx = 0;
        static LL ps[N * 2], h[N * 2];
        int cnt = circle.size();
        for (int i = 0; i < cnt; i++)
        {
            int u = circle[i];
            h[i] = getHeight(u, mx);
            ps[i] = sv[u];
        }
        std::copy(h, h + cnt, h + cnt);
        for (int i = 0; i < cnt; i++) ps[i + cnt] = ps[cnt - 1] + ps[i];

        static int Q[N * 2];
        int hh = 0, tt = 0;
        for (int i = 0; i < cnt * 2; i++)
        {
            if (hh != tt && i - Q[hh] + 1 > cnt) hh++;
            if (hh != tt) mx = std::max(mx, ps[i] + h[i] - ps[Q[hh]] + h[Q[hh]]);
            while (hh != tt && h[Q[tt - 1]] - ps[Q[tt - 1]] <= h[i] - ps[i]) tt--;
            Q[tt++] = i;
        }

        res += mx;
    }
    
    cout << res << '\n';

    return 0;
}
```
