### Splay

题目：https://www.acwing.com/problem/content/2439/

题解：https://www.acwing.com/solution/content/140676/

```c++
#include <cstdio>
#include <cstring>
#include <iostream>
#include <algorithm>

using LL = long long;
using std::cin, std::cout;

const int N = 1e5 + 5;

int n, m;

struct Node
{
    int s[2], p, v;
    int cnt, flag;

    void init(int _v, int _p)
    {
        v = _p, p = _p;
        cnt = 1;
    }
} tr[N];
int root, idx;

void pushup(int u)
{
    tr[u].cnt = tr[tr[u].s[0]].cnt + tr[tr[u].s[1]].cnt + 1;
}

void pushdown(int u)
{
    if (tr[u].flag)
    {
        std::swap(tr[u].s[0], tr[u].s[1]);
        tr[tr[u].s[0]].flag ^= 1;
        tr[tr[u].s[1]].flag ^= 1;
        tr[u].flag = 0;
    }
}

void rotate(int x)
{
    int y = tr[x].p, z = tr[y].p;
    int k = tr[y].s[1] == x;

    tr[z].s[tr[z].s[1] == y] = x, tr[x].p = z;
    tr[y].s[k] = tr[x].s[k ^ 1], tr[tr[x].s[k ^ 1]].p = y;
    tr[x].s[k ^ 1] = y, tr[y].p = x;

    pushup(y), pushup(x);
}

void splay(int x, int k)
{
    while (tr[x].p != k)
    {
        int y = tr[x].p, z = tr[y].p;
        if (z != k)
        {
            if ((tr[y].s[1] == x) ^ (tr[z].s[1] == y)) rotate(x);
            else rotate(y);
        }
        rotate(x);
    }
    if (k == 0) root = x;
}

void insert(int v)
{
    int u = root, p = 0;
    while (u)
    {
        p = u;
        u = tr[u].s[v > tr[u].v];
    }
    u = ++idx;
    if (p) tr[p].s[v > tr[p].v] = u;
    tr[u].init(v, p);
    splay(u, 0);
}

int getK(int k)
{
    int u = root;
    while (true)
    {
        pushdown(u);
        if (tr[tr[u].s[0]].cnt >= k) u = tr[u].s[0];
        else if (tr[tr[u].s[0]].cnt + 1 == k) return u;
        else
        {
            k -= tr[tr[u].s[0]].cnt + 1;
            u = tr[u].s[1];
        }
    }
    return -1;
}

void print(int u)
{
    pushdown(u);
    if (tr[u].s[0]) print(tr[u].s[0]);
    if (tr[u].v >= 1 && tr[u].v <= n) cout << tr[u].v << ' ';
    if (tr[u].s[1]) print(tr[u].s[1]);
}

int main()
{
    std::ios::sync_with_stdio(false);
    cin.tie(nullptr), cout.tie(nullptr);
    
    cin >> n >> m;
    for (int i = 0; i <= n + 1; i++) insert(i);
    for (int i = 0; i < m; i++)
    {
        int l, r; cin >> l >> r;
        l++, r++;
        l = getK(l - 1), r = getK(r + 1);
        splay(l, 0), splay(r, l);
        tr[tr[r].s[0]].flag ^= 1;
    }
    
    print(root);

    return 0;
}
```

### 郁闷的出纳员

题目：https://www.acwing.com/problem/content/952/

题解：https://www.acwing.com/solution/content/140833/

```c++
#include <cstdio>
#include <cstring>
#include <iostream>
#include <algorithm>

using LL = long long;
using std::cin, std::cout;

const int N = 1e5 + 5, INF = 1e8 + 5;

int n, mn, delta;

struct Node
{
    int s[2], p, v;
    int cnt;

    void init(int _v, int _p)
    {
        v = _v, p = _p;
        cnt = 1;
    }
} tr[N];
int root, idx;

void pushup(int u)
{
    tr[u].cnt = tr[tr[u].s[0]].cnt + tr[tr[u].s[1]].cnt + 1;
}

void rotate(int x)
{
    int y = tr[x].p, z = tr[y].p;
    int k = tr[y].s[1] == x;

    tr[z].s[tr[z].s[1] == y] = x, tr[x].p = z;
    tr[y].s[k] = tr[x].s[k ^ 1], tr[tr[x].s[k ^ 1]].p = y;
    tr[x].s[k ^ 1] = y, tr[y].p = x;
    pushup(y), pushup(x);  // 顺序不可以颠倒
}

void splay(int x, int k)
{
    while (tr[x].p != k)
    {
        int y = tr[x].p, z = tr[y].p;
        if (z != k)
        {
            if ((tr[y].s[1] == x) ^ (tr[z].s[1] == y)) rotate(x);
            else rotate(y);
        }
        rotate(x);
    }
    if (k == 0) root = x;
}

int insert(int v)
{
    int p = 0, u = root;
    while (u)
    {
        p = u;
        u = tr[u].s[v > tr[u].v];
    }
    u = ++idx;
    if (p) tr[p].s[v > tr[p].v] = u;
    tr[u].init(v, p);
    splay(u, 0);

    return u;
}

int get(int v)
{
    int u = root, res = -1;
    while (u)
    {
        if (tr[u].v >= v) res = u, u = tr[u].s[0];
        else u = tr[u].s[1];
    }
    return res;
}

int getK(int k)
{
    int u = root;
    while (u)
    {
        if (tr[tr[u].s[0]].cnt >= k) u = tr[u].s[0];
        else if (tr[tr[u].s[0]].cnt + 1 == k) return u;
        else
        {
            k -= tr[tr[u].s[0]].cnt + 1;
            u = tr[u].s[1];
        }
    }

    return -1;
}

int main()
{
    std::ios::sync_with_stdio(false);
    cin.tie(nullptr), cout.tie(nullptr);

    int L = insert(-INF), R = insert(INF);

    cin >> n >> mn;
    int tot = 0;
    for (int i = 0; i < n; i++)
    {
        char op[2];
        int k;
        cin >> op >> k;
        if (op[0] == 'I')
        {
            if (k >= mn)
            {
                k -= delta;
                insert(k);
                tot++;
            }
        }
        else if (op[0] == 'A')
        {
            delta += k;
        }
        else if (op[0] == 'S')
        {
            delta -= k;
            R = get(mn - delta);
            splay(R, 0), splay(L, R);
            tr[L].s[1] = 0;
            pushup(L), pushup(R);
        }
        else if (op[0] == 'F')
        {
            if (k > tr[root].cnt - 2) cout << -1 << '\n';
            else
            {
                int u = getK(tr[root].cnt - k);
                cout << tr[u].v + delta << '\n';
            }
        }
    }

    cout << tot - (tr[root].cnt - 2) << '\n';

    return 0;
}
```

### 永无乡

题目：https://www.acwing.com/problem/content/description/1065/

题解：

https://www.acwing.com/solution/content/142368/

https://www.acwing.com/solution/content/283270/

```c++
#include <cstdio>
#include <cstring>
#include <iostream>
#include <algorithm>

using LL = long long;
using std::cin, std::cout;

const int N = 1e5 + 5, M = 17 * N;

int n, m;
struct Node
{
    int s[2], p, v, id;
    int cnt;

    void init(int _p, int _v, int _id)
    {
        p = _p, v = _v, id = _id;
        cnt = 1;
    }
} tr[M];
int root[N], idx;

int P[N];

int find(int x)
{
    if (x != P[x]) P[x] = find(P[x]);
    return P[x];
}

void pushup(int x)
{
    tr[x].cnt = tr[tr[x].s[0]].cnt + tr[tr[x].s[1]].cnt + 1;
}

void rotate(int x)
{
    int y = tr[x].p, z = tr[y].p;
    int k = tr[y].s[1] == x;

    tr[z].s[tr[z].s[1] == y] = x, tr[x].p = z;
    tr[y].s[k] = tr[x].s[k ^ 1], tr[tr[x].s[k ^ 1]].p = y;
    tr[x].s[k ^ 1] = y, tr[y].p = x;
    pushup(y), pushup(x);
}

void splay(int& root, int x, int k)
{
    while (tr[x].p != k)
    {
        int y = tr[x].p, z = tr[y].p;
        if (z != k)
        {
            if ((tr[y].s[1] == x) ^ (tr[z].s[1] == y)) rotate(x);
            else rotate(y);
        }
        rotate(x);
    }
    if (k == 0) root = x;
}

void insert(int& root, int v, int id)
{
    int p = 0, u = root;
    while (u)
    {
        p = u;
        u = tr[u].s[v > tr[u].v];
    }
    u = ++idx;
    if (p) tr[p].s[v > tr[p].v] = u;
    tr[u].init(p, v, id);
    splay(root, u, 0);
}

void merge(int& root, int x)  // 把x子树合并到root
{
    if (tr[x].s[0]) merge(root, tr[x].s[0]);
    if (tr[x].s[1]) merge(root, tr[x].s[1]);

    insert(root, tr[x].v, tr[x].id);
}

int getK(int u, int k)
{
    while (u)
    {
        int left = tr[tr[u].s[0]].cnt;
        if (left >= k) u = tr[u].s[0];
        else if (left + 1 == k) return u;
        else
        {
            k -= left + 1;
            u = tr[u].s[1];
        }
    }
    return -1;
}

int main()
{
    std::ios::sync_with_stdio(false);
    cin.tie(nullptr), cout.tie(nullptr);

    cin >> n >> m;
    for (int i = 1; i <= n; i++)
    {
        P[i] = root[i] = i;
        int v; cin >> v;
        tr[i].init(0, v, i);
    }
    idx = n;

    for (int i = 0; i < m; i++)
    {
        int a, b; cin >> a >> b;
        a = find(a), b = find(b);
        if (a != b)
        {
            if (tr[root[a]].cnt > tr[root[b]].cnt) std::swap(a, b);
            merge(root[b], root[a]);
            P[a] = b;
        }
    }

    int q; cin >> q;
    while (q--)
    {
        char op[2]; cin >> op;
        int a, b; cin >> a >> b;
        if (op[0] == 'B')
        {
            a = find(a), b = find(b);
            if (a != b)
            {
                if (tr[root[a]].cnt > tr[root[b]].cnt) std::swap(a, b);
                merge(root[b], root[a]);
                P[a] = b;
            }
        }
        else
        {
            a = root[find(a)];
            if (tr[a].cnt >= b)
            {
                int u = getK(a, b);
                cout << tr[u].id << '\n';
            }
            else cout << -1 << '\n';
        }
    }

    return 0;
}
```

### 维护数列

题目：https://www.acwing.com/problem/content/description/957/

题解：https://www.acwing.com/solution/content/143117/

```c++
#include <cstdio>
#include <cstring>
#include <iostream>
#include <algorithm>

using LL = long long;
using std::cin, std::cout;

const int N = 5e5 + 5, INF = 1e8;

int n, m;
struct Node
{
	int s[2], p, v, cnt;
	int sum, ms, ls, rs;
	bool rev, same;

	void init(int _v, int _p)
	{
		s[0] = s[1] = 0;
		v = _v, p = _p;
		cnt = 1;
		ms = sum = v;
		ls = rs = std::max(v, 0);
		rev = same = false;
	}
} tr[N];
int root;
int empty[N], top;  // 垃圾回收

void gc(int x)
{
	if (tr[x].s[0]) gc(tr[x].s[0]);
	if (tr[x].s[1]) gc(tr[x].s[1]);
	empty[++top] = x;
}

void pushup(int x)
{
	auto& u = tr[x], & l = tr[u.s[0]], & r = tr[u.s[1]];
	u.cnt = l.cnt + r.cnt + 1;
	u.sum = l.sum + r.sum + u.v;
	u.ls = std::max(l.ls, l.sum + u.v + r.ls);
	u.rs = std::max(r.rs, r.sum + u.v + l.rs);
	u.ms = std::max({ l.ms, r.ms, l.rs + u.v + r.ls });
}

void pushdown(int x)
{
	auto& u = tr[x], & l = tr[u.s[0]], & r = tr[u.s[1]];
	if (u.same)
	{
		u.same = u.rev = false;
		if (u.s[0]) l.same = true, l.v = u.v, l.sum = l.v * l.cnt;
		if (u.s[1]) r.same = true, r.v = u.v, r.sum = r.v * r.cnt;
		if (u.v > 0)
		{
			if (u.s[0]) l.ms = l.ls = l.rs = l.sum;
			if (u.s[1]) r.ms = r.ls = r.rs = r.sum;
		}
		else
		{
			if (u.s[0]) l.ms = l.v, l.ls = l.rs = 0;
			if (u.s[1]) r.ms = r.v, r.ls = r.rs = 0;
		}
	}
	else if (u.rev)
	{
		u.rev = false, l.rev ^= true, r.rev ^= true;
		std::swap(l.ls, l.rs), std::swap(l.s[0], l.s[1]);
		std::swap(r.ls, r.rs), std::swap(r.s[0], r.s[1]);
	}
}

void rotate(int x)
{
	int y = tr[x].p, z = tr[y].p;
	int k = tr[y].s[1] == x;

	tr[z].s[tr[z].s[1] == y] = x, tr[x].p = z;
	tr[y].s[k] = tr[x].s[k ^ 1], tr[tr[x].s[k ^ 1]].p = y;
	tr[y].p = x, tr[x].s[k ^ 1] = y;

	pushup(y), pushup(x);
}

void splay(int x, int k)
{
	while (tr[x].p != k)
	{
		int y = tr[x].p, z = tr[y].p;
		if (z != k)
		{
			if ((tr[y].s[1] == x) ^ (tr[z].s[1] == y)) rotate(x);
			else rotate(y);
		}
		rotate(x);
	}
	if (k == 0) root = x;
}

int getK(int k)
{
	int u = root;
	while (u)
	{
		pushdown(u);
		if (tr[tr[u].s[0]].cnt >= k) u = tr[u].s[0];
		else if (tr[tr[u].s[0]].cnt + 1 == k) return u;
		else
		{
			k -= tr[tr[u].s[0]].cnt + 1;
			u = tr[u].s[1];
		}
	}
	return -1;
}

int A[N];

int build(int l, int r, int p)
{
	if (l > r) return 0;

	int mid = l + r >> 1;
	int u = empty[top--];
	tr[u].init(A[mid], p);
	tr[u].s[0] = build(l, mid - 1, u);
	tr[u].s[1] = build(mid + 1, r, u);
	pushup(u);

	return u;
}

int main()
{
	std::ios::sync_with_stdio(false);
	cin.tie(nullptr), cout.tie(nullptr);
	
	tr[0].ms = -INF;  // 便于子节点为空时的pushup
	
	for (int i = 1; i < N; i++) empty[++top] = i;

	cin >> n >> m;
	for (int i = 1; i <= n; i++) cin >> A[i];
	A[0] = A[n + 1] = -INF;  // 不影响答案tr[root].ms
	root = build(0, n + 1, 0);

	while (m--)
	{
		std::string op; cin >> op;
		if (op == "INSERT")
		{
			int posi, tot; cin >> posi >> tot;
			for (int i = 1; i <= tot; i++) cin >> A[i];
			int l = getK(posi + 1), r = getK(posi + 2);
			splay(l, 0), splay(r, l);
			int u = build(1, tot, r);
			tr[r].s[0] = u;
			pushup(r), pushup(l);
		}
		else if (op == "DELETE")
		{
			int posi, tot; cin >> posi >> tot;
			int l = getK(posi), r = getK(posi + 1 + tot);
			splay(l, 0), splay(r, l);
			gc(tr[r].s[0]);
			tr[r].s[0] = 0;
			pushup(r), pushup(l);
		}
		else if (op == "MAKE-SAME")
		{
			int posi, tot, c; cin >> posi >> tot >> c;
			int l = getK(posi), r = getK(posi + 1 + tot);
			splay(l, 0), splay(r, l);
			auto& u = tr[tr[r].s[0]];
			u.same = true;
			u.v = c, u.sum = tot * c;
			if (c > 0) u.ms = u.ls = u.rs = u.sum;
			else u.ms = c, u.ls = u.rs = 0;
			pushup(r), pushup(l);
		}
		else if (op == "REVERSE")
		{
			int posi, tot; cin >> posi >> tot;
			int l = getK(posi), r = getK(posi + 1 + tot);
			splay(l, 0), splay(r, l);
			auto& u = tr[tr[r].s[0]];
			u.rev ^= true;
			std::swap(u.ls, u.rs), std::swap(u.s[0], u.s[1]);
			pushup(r), pushup(l);
		}
		else if (op == "GET-SUM")
		{
			int posi, tot; cin >> posi >> tot;
			int l = getK(posi), r = getK(posi + 1 + tot);
			splay(l, 0), splay(r, l);
			cout << tr[tr[r].s[0]].sum << '\n';
		}
		else
		{
			cout << tr[root].ms << '\n';
		}
	}

	return 0;
}
```

## 树套树

### 树套树-简单版

题目：https://www.acwing.com/problem/content/description/2490/

题解：https://www.acwing.com/solution/content/155519/

要在O2优化下才能通过

```c++
#include <cstdio>
#include <cstring>
#include <iostream>
#include <algorithm>
#include <set>

using LL = long long;
using std::cin, std::cout;

const int N = 5e4 + 5, INF = 1e9;

int n, m;
struct STNode
{
    int l, r;
    std::multiset<int> S;
} tr[N * 4];
int A[N];

void build(int u, int l, int r)
{
    tr[u].l = l, tr[u].r = r;
    auto& S = tr[u].S;
    S.insert(-INF), S.insert(INF);
    for (int i = l; i <= r; i++) S.insert(A[i]);
    if (l < r)
    {
        int mid = (l + r) / 2;
        build(u * 2, l, mid), build(u * 2 + 1, mid + 1, r);
    }
}

void update(int u, int p, int x)
{
    auto& S = tr[u].S;
    S.erase(S.find(A[p]));
    S.insert(x);
    if (tr[u].l < tr[u].r)
    {
        int mid = (tr[u].l + tr[u].r) / 2;
        if (p <= mid) update(u * 2, p, x);
        else update(u * 2 + 1, p, x);
    }
}

int query(int u, int l, int r, int x)
{
    if (l <= tr[u].l && r >= tr[u].r)
    {
        auto& S = tr[u].S;
        auto it = S.lower_bound(x);
        --it;
        return *it;
    }
    else
    {
        int mid = (tr[u].l + tr[u].r) / 2;
        int res = -INF;
        if (l <= mid) res = std::max(res, query(u * 2, l, r, x));
        if (r > mid) res = std::max(res, query(u * 2 + 1, l, r, x));
        return res;
    }
}

int main()
{
    std::ios::sync_with_stdio(false);
    cin.tie(nullptr), cout.tie(nullptr);

    cin >> n >> m;
    for (int i = 1; i <= n; i++) cin >> A[i];
    build(1, 1, n);

    while (m--)
    {
        int op; cin >> op;
        if (op == 1)
        {
            int p, x; cin >> p >> x;
            update(1, p, x);
            A[p] = x;
        }
        else
        {
            int l, r, x; cin >> l >> r >> x;
            cout << query(1, l, r, x) << '\n';
        }
    }

    return 0;
}
```

```c++
    }

    void insert(int& root, int v)
    {
        int p = 0, u = root;
        while (u)
        {
            p = u;
            u = tr[u].s[v > tr[u].v];
        }
        u = ++idx;
        if (p) tr[p].s[v > tr[p].v] = u;
        tr[u].init(v, p);
        splay(root, u, 0);
    }

    void delOne(int& root, int v)  // 删除一个=v的节点
    {
        int u = root;
        while (u)
        {
            if (tr[u].v == v) break;
            u = tr[u].s[v > tr[u].v];
        }
        splay(root, u, 0);
        int l = tr[u].s[0], r = tr[u].s[1];
        while (tr[l].s[1]) l = tr[l].s[1];
        while (tr[r].s[0]) r = tr[r].s[0];
        splay(root, l, 0), splay(root, r, l);
        tr[r].s[0] = 0;
        pushup(r), pushup(l);
    }

    int get(int& root, int v)  // 找到 < v的节点个数
    {
        int res = 0, u = root;
        while (u)
        {
            if (tr[u].v < v)
            {
                res += tr[tr[u].s[0]].cnt + 1;
                u = tr[u].s[1];
            }
            else u = tr[u].s[0];
        }
        return res;
    }

    int getPre(int& root, int v)  // 找到v的前驱
    {
        int u = root, res = -INF;
        while (u)
        {
            if (tr[u].v < v)
            {
                res = std::max(res, tr[u].v);
                u = tr[u].s[1];
            }
            else u = tr[u].s[0];
        }
        return res;
    }

    int getNxt(int& root, int v)  // 找到v的后继
    {
        int u = root, res = INF;
        while (u)
        {
            if (tr[u].v > v)
            {
                res = std::min(res, tr[u].v);
                u = tr[u].s[0];
            }
            else u = tr[u].s[1];
        }
        return res;
    }
}
```

```c++
{
    int op, a, b, c;

std::vector<int> disc;

int find(int x)
{
    return std::lower_bound(disc.begin(), disc.end(), x) - disc.begin();
}

struct node  // 内层线段树节点
{
    int l, r;
    LL sum, add;
} tr[S];
int idx;

struct Node
{
    int l, r;
    int root;  // 内层线段树根节点指针
} TR[M];

void build(int u, int l, int r)  // 建立外层线段树
{
    TR[u] = { l, r, ++idx };
    if (l < r)
    {
        int mid = (l + r) / 2;
        build(u * 2, l, mid), build(u * 2 + 1, mid + 1, r);
    }
}

int overlap(int a, int b, int c, int d)
{
    return std::min(b, d) - std::max(a, c) + 1;
}

void increment(int u, int l, int r, int L, int R)  // 内层线段树[L, R]加1
{
    tr[u].sum += overlap(l, r, L, R);

    if (l >= L && r <= R)
    {
        tr[u].add++;
    }
    else
    {
        int mid = (l + r) / 2;
        if (L <= mid)
        {
            if (tr[u].l == 0) tr[u].l = ++idx;
            increment(tr[u].l, l, mid, L, R);
        }
        if (R > mid)
        {
            if (tr[u].r == 0) tr[u].r = ++idx;
            increment(tr[u].r, mid + 1, r, L, R);
        }
    }
}

LL getSum(int u, int l, int r, int L, int R, LL add)
{
    if (l >= L && r <= R) return tr[u].sum + add * (r - l + 1);

    LL res = 0;
    add += tr[u].add;
    int mid = (l + r) / 2;
    if (L <= mid)
    {
        if (tr[u].l) res += getSum(tr[u].l, l, mid, L, R, add);
        else res += add * overlap(l, mid, L, R);
    }
    if (R > mid)
    {
        if (tr[u].r) res += getSum(tr[u].r, mid + 1, r, L, R, add);
        else res += add * overlap(mid + 1, r, L, R);
    }
    return res;
}

void update(int u, int l, int r, int c)
{
    increment(TR[u].root, 1, n, l, r);

    if (TR[u].l < TR[u].r)
    {
        int mid = (TR[u].l + TR[u].r) / 2;
        if (c <= mid) update(u * 2, l, r, c);
        else update(u * 2 + 1, l, r, c);
    }
}

int queryK(int u, int l, int r, int k)
{
    if (TR[u].l == TR[u].r) return TR[u].l;

    LL rtSum = getSum(TR[u * 2 + 1].root, 1, n, l, r, 0);
    if (k <= rtSum) return queryK(u * 2 + 1, l, r, k);
    else return queryK(u * 2, l, r, k - rtSum);
}

{

    cin.tie(nullptr), cout.tie(nullptr);

    cin >> n >> m;
    for (int i = 0; i < m; i++)
    {
        int op, a, b, c; cin >> op >> a >> b >> c;
        OP[i] = { op, a, b, c };
        if (op == 1) disc.push_back(c);
    }

    std::sort(disc.begin(), disc.end());
    disc.erase(std::unique(disc.begin(), disc.end()), disc.end());
    build(1, 0, disc.size() - 1);

    for (int i = 0; i < m; i++)
    {
        auto [op, a, b, c] = OP[i];
        if (op == 1)
        {
            update(1, a, b, find(c));
        }
        else
        {
            cout << disc[queryK(1, a, b, c)] << '\n';
        }
    }

    return 0;
}
```

```c++
#include <cstdio>
#include <cstring>
#include <cmath>
#include <iostream>
#include <algorithm>

using LL = long long;
using std::cin, std::cout;

const int N = 1e5 + 5, M = 330;

int n, m, len;
int A[N];
LL sum[M], add[M];

int get(int x)  // 返回块的索引
{
    return x / len;
}

void update(int l, int r, int d)
{
    if (get(l) == get(r))  // 在一个块内，直接暴力
    {
        sum[get(l)] += (r - l + 1) * d;
        for (int i = l; i <= r; i++) A[i] += d;
    }
    else
    {
        int i = l, j = r;
        for (; get(i) == get(l); i++) A[i] += d;
        sum[get(l)] += (i - l) * d;
        for (; get(j) == get(r); j--) A[j] += d;
        sum[get(r)] += (r - j) * d;
        for (int k = get(i); k <= get(j); k++)
        {
            sum[k] += d * len;
            add[k] += d;
        }
    }
}

LL query(int l, int r)
{
    LL res = 0;
    if (get(l) == get(r))
    {
        for (int i = l; i <= r; i++) res += A[i];
        res += add[get(l)] * (r - l + 1);
    }
    else
    {
        int i = l, j = r;
        for (; get(i) == get(l); i++) res += A[i];
        res += add[get(l)] * (i - l);
        for (; get(j) == get(r); j--) res += A[j];
        res += add[get(r)] * (r - j);
        for (int k = get(i); k <= get(j); k++) res += sum[k];
    }

    return res;
}
```

```c++
        {
            k--;
            buf[len++] = c;
        }
    }

}

void move(int k)
{
    int cur = L[0].r; 
    while (k > L[cur].len)
    {
        k -= L[cur].len;
        cur = L[cur].r;
    }
    x = cur, y = k - 1;
}

void add(int a, int b)  // 在a节点之后插入b节点
{
    L[b].r = L[a].r, L[L[b].r].l = b;
    L[a].r = b, L[b].l = a;
}

void insert(int k)  // 在光标之后插入k个字符
{
    if (y < L[x].len - 1)
    {
        int u = empty[top--], len = L[x].len - 1 - y;
        memcpy(L[u].S, L[x].S + y + 1, len * sizeof(char));
        L[u].len = len, L[x].len = y + 1;
        add(x, u);
    }

    int pre = x, cur = 0, len = 0;
    for (int i = 0; i < k; i += len)
    {
        cur = empty[top--], len = std::min(N, k - i);
        memcpy(L[cur].S, buf + i, len * sizeof(char));
        L[cur].len = len;
        add(pre, cur);
        pre = cur;
    }
}

void del(int k)
{
    if (k <= L[x].len - 1 - y)
    {
        for (int i = y + 1 + k; i < L[x].len; i++)  // 不要用memcpy
        {
            L[x].S[i - k] = L[x].S[i];
        }
        L[x].len -= k;
    }
    else
    {
        k -= L[x].len - 1 - y;
        L[x].len = y + 1;
        int tail = L[x].r;
        while (tail && k >= L[tail].len)
        {
            k -= L[tail].len;
            empty[++top] = tail;
            tail = L[tail].r;
        }
        if (tail)
        {
            if (k > 0)
            {
                for (int i = k; i < L[tail].len; i++)  // 不要用memcpy
                {
                    L[tail].S[i - k] = L[tail].S[i];
                }
                L[tail].len -= k;
            }
        }
        L[x].r = tail, L[tail].l = x;
    }
}

{
    int len = 0;
    if (k <= L[x].len - 1 - y)
    {
        for (int i = 1; i <= k; i++) putchar(L[x].S[y + i]);
    }
    else
    {
        k -= L[x].len - 1 - y;
        for (int i = y + 1; i < L[x].len; i++) putchar(L[x].S[i]);

        int cur = L[x].r;
        while (cur && k >= L[cur].len)
        {
            k -= L[cur].len;
            for (int i = 0; i < L[cur].len; i++) putchar(L[cur].S[i]);
            cur = L[cur].r;
        }
        if (cur) 
        {
            for (int i = 0; i < k; i++) putchar(L[cur].S[i]);
        }
    }

                y += L[i].len;
            }

            L[i].r = L[j].r, L[L[j].r].l = i;
            empty[++top] = j;
        }
    }
}

    {
```

```c++
	        if (ar != br) return ar < br;
	        return a.t < b.t;
	    });

	for (int k = 1, i = 1, j = 0, cur = 0, tmp = 0; k <= mq; k++)
	{
		auto [l, r, t, id] = Q[k];
		while (j < r) add(A[++j], tmp);
		while (j > r) del(A[j--], tmp);
		while (i < l) del(A[i++], tmp);
		while (i > l) add(A[--i], tmp);
		while (cur < t)
		{
			cur++;
			if (R[cur].p >= l && R[cur].p <= r)
			{
				del(A[R[cur].p], tmp);
				add(R[cur].c, tmp);
			}
			std::swap(A[R[cur].p], R[cur].c);
		}
		while (cur > t)
		{
			if (R[cur].p >= l && R[cur].p <= r)
			{
				del(A[R[cur].p], tmp);
				add(R[cur].c, tmp);
			}
			std::swap(A[R[cur].p], R[cur].c);
			cur--;
		}
		res[id] = tmp;
	}

	for (int i = 1; i <= mq; i++) cout << res[i] << '\n';
	return 0;
}
```

### 回滚莫队

#### 历史研究

题目：https://www.acwing.com/problem/content/2525/

```c++
#include <cstring>
#include <cmath>
#include <iostream>
#include <algorithm>
#include <vector>

using LL = long long;
using std::cin, std::cout;

const int N = 1e5 + 5;

int n, m;

{
    int id, l, r;

{
    return std::lower_bound(disc.begin(), disc.end(), num) - disc.begin();

{
    return x / len;

{

        Q[i] = { i, l, r };
    }

    std::sort(Q, Q + m,
        [](const Query& a, const Query& b)
        {
            int i = get(a.l), j = get(b.l);
            if (i != j) return i < j;
            return a.r < b.r;
        });

    for (int x = 0; x < m;)
    {
        int y = x, right = get(Q[x].l) * len + len - 1;
        while (y < m && get(Q[x].l) == get(Q[y].l)) y++;
        
        for (; x < y && Q[x].r <= right; x++)
        {
            LL tmp = 0;
            auto [id, l, r] = Q[x];
            for (int k = l; k <= r; k++) add(A[k], tmp);
            res[id] = tmp;
            for (int k = l; k <= r; k++) cnt[A[k]]--;

        {

            while (i > l) add(A[--i], tmp);
```

### 树上莫队

#### 树上计数2

题目：https://www.acwing.com/problem/content/description/2536/

题解：https://www.acwing.com/solution/content/157759/

```c++
#include <cstdio>
#include <cstring>
#include <cmath>
#include <iostream>
#include <algorithm>
#include <vector>

using LL = long long;
using std::cin, std::cout;

const int N = 4e4 + 5, M = 1e5 + 5;

int n, m;
int A[N];
std::vector<int> disc;

int find(int num)
{
	return std::lower_bound(disc.begin(), disc.end(), num) - disc.begin();
}

int H[N];
struct EdgeNode
{
	int v, n;
} E[N * 2];
int idx;

void addEdge(int a, int b)
{
	E[++idx] = { b, H[a] }, H[a] = idx;
}

int seq[N * 2], top;
int first[N], last[N];

void dfs(int u, int fa)  // 获得欧拉序
{
	seq[++top] = u;
	first[u] = top;
	for (int i = H[u]; i; i = E[i].n)
	{
		int v = E[i].v;
		if (v != fa) dfs(v, u);
	}
	seq[++top] = u;
	last[u] = top;
}

int depth[N], father[N][16];

void bfs()
{
	static int Q[N];
	int hh = 0, tt = 0;
	Q[tt++] = 1;

	depth[1] = 1;

	while (hh != tt)
	{
		int u = Q[hh++];
		for (int i = H[u]; i; i = E[i].n)
		{
			int v = E[i].v;
			if (depth[v] == 0)
			{
				depth[v] = depth[u] + 1;
				father[v][0] = u;
				for (int k = 1; k <= 15; k++)
				{
					father[v][k] = father[father[v][k - 1]][k - 1];
				}
				Q[tt++] = v;
			}
		}
	}
}

int lca(int a, int b)
{
	if (depth[a] < depth[b]) std::swap(a, b);

	for (int k = 15; k >= 0; k--)
		if (depth[father[a][k]] >= depth[b])
			a = father[a][k];

	if (a == b) return a;

	for (int k = 15; k >= 0; k--)
		if (father[a][k] != father[b][k])
		{
			a = father[a][k];
			b = father[b][k];
		}

	return father[a][0];
}

int cnt[N];
struct Query
{
	int id, l, r, p;
} Q[M];
int res[M];

int len;

int get(int x)
{
	return x / len;
}

bool st[N];

void add(int x, int& res)
{
	st[x] ^= true;
	if (st[x])
	{
		if (cnt[A[x]]++ == 0) res++;
	}
	else
	{
		if (--cnt[A[x]] == 0) res--;
	}
}

int main()
{
	std::ios::sync_with_stdio(false);
	cin.tie(nullptr), cout.tie(nullptr);

	cin >> n >> m;
	for (int i = 1; i <= n; i++)
	{
		cin >> A[i];
		disc.push_back(A[i]);
	}
	std::sort(disc.begin(), disc.end());
	for (int i = 1; i <= n; i++) A[i] = find(A[i]);

	for (int i = 0; i < n - 1; i++)
	{
		int a, b; cin >> a >> b;
		addEdge(a, b), addEdge(b, a);
	}

	dfs(1, -1);

	bfs();  // lca的预处理

	for (int i = 0; i < m; i++)
	{
		int a, b; cin >> a >> b;
		if (first[a] > first[b]) std::swap(a, b);
		int p = lca(a, b);
		if (a == p) Q[i] = { i, first[a], first[b], 0 };
		else Q[i] = { i, last[a], first[b], p };
	}

	len = sqrt(top);
	std::sort(Q, Q + m,
		[](const Query& a, const Query& b)
		{
			int al = get(a.l), bl = get(b.l);
			if (al != bl) return al < bl;
			return a.r < b.r;
		});

	for (int k = 0, i = 1, j = 0, tmp = 0; k < m; k++)
	{
		auto [id, l, r, p] = Q[k];
		while (j < r) add(seq[++j], tmp);
		while (j > r) add(seq[j--], tmp);
		while (i < l) add(seq[i++], tmp);
		while (i > l) add(seq[--i], tmp);
		if (p) add(p, tmp);
		res[id] = tmp;
		if (p) add(p, tmp);
	}

	for (int i = 0; i < m; i++) cout << res[i] << '\n';

	return 0;
}
```

### 二次离线莫队

#### 二次离线莫队

题目：https://www.acwing.com/problem/content/description/2537/

题解：https://www.acwing.com/solution/content/158344/

```c++
#include <cstdio>
#include <cstring>
#include <cmath>
#include <iostream>
#include <algorithm>
#include <vector>

using LL = long long;
using std::cin, std::cout;

const int N = 1e5 + 5;

int n, m, K;
int A[N];
int len;
struct Query
{
    int id, l, r;
    LL res;
} Q[N];

int get(int x)
{
    return x / len;
}

struct Range
{
    int x, l, r, t;
};
std::vector<Range> R[N];

std::vector<int> kBits;

int countBit(int a)
{
    int cnt = 0;
    while (a)
    {
        a -= a & -a;  // lowbit
        cnt++;
    }
    return cnt;
}

int f[N], g[N];  // g[x]表示前i个数中与x配对的个数，f[i]表示A[i + 1]与前i个数配对的个数

LL res[N];

int main()
{
    std::ios::sync_with_stdio(false);
    cin.tie(nullptr), cout.tie(nullptr);

    cin >> n >> m >> K;
    for (int i = 1; i <= n; i++) cin >> A[i];
    for (int i = 0; i < m; i++)
    {
        int l, r; cin >> l >> r;
        Q[i] = { i, l, r };
    }

    len = sqrt(n);

    std::sort(Q, Q + m,
        [](const Query& a, const Query& b)
        {
            int i = get(a.l), j = get(b.l);
            if (i != j) return i < j;
            return a.r < b.r;
        });

    for (int i = 0; i < 1 << 14; i++)
        if (countBit(i) == K)
            kBits.push_back(i);

    for (int i = 1; i <= n; i++)
    {
        for (int kb : kBits) g[kb ^ A[i]]++;
        f[i] = g[A[i + 1]];
    }
    
    
    for (int k = 0, i = 1, j = 0; k < m; k++)
    {
        auto& [id, l, r, res] = Q[k];
        if (j < r) R[i - 1].push_back({ k, j + 1, r, -1 });
        while (j < r) res += f[j++];
        if (j > r) R[i - 1].push_back({ k, r + 1, j, 1 });
        while (j > r) res -= f[--j];
        if (i < l) R[j].push_back({ k, i, l - 1, -1 });
        while (i < l) res += f[i - 1] + !K, i++;  // 这里是大写K
        if (i > l) R[j].push_back({ k, l, i - 1, 1 });
        while (i > l) res -= f[i - 2] + !K, i--;
    }

    memset(g, 0, sizeof g);

    for (int i = 1; i <= n; i++)
    {
        for (int kb : kBits) g[kb ^ A[i]]++;
        for (auto [x, l, r, t] : R[i])
        {
            for (int j = l; j <= r; j++) Q[x].res += g[A[j]] * t;
        }
    }
    for (int i = 1; i < m; i++) Q[i].res += Q[i - 1].res;  // 必须要累加

    for (int i = 0; i < m; i++) res[Q[i].id] = Q[i].res;
    for (int i = 0; i < m; i++) cout << res[i] << '\n';

    return 0;
}
```

## 树链剖分

### 树链剖分

题目：https://www.acwing.com/problem/content/2570/

题解：https://www.acwing.com/solution/content/158813/

```c++
#include <cstdio>
#include <cstring>
#include <iostream>
#include <algorithm>

using LL = long long;
using std::cin, std::cout;

const int N = 1e5 + 5, M = N * 2;

int n, m;
int A[N];
int H[N];
struct EdgeNode
{
    int v, n;
} E[M];
int idx;

void add(int a, int b)
{
    E[++idx] = { b, H[a] }, H[a] = idx;
}

int size[N], father[N], son[N], depth[N];

void dfs1(int u, int fa)
{
    size[u] = 1;
    father[u] = fa;
    if (fa == -1) depth[u] = 1;
    else depth[u] = depth[fa] + 1;

    for (int i = H[u]; i; i = E[i].n)
    {
        int v = E[i].v;
        if (v != fa)
        {
            dfs1(v, u);
            size[u] += size[v];
            if (son[u] == 0 || size[v] > size[son[u]])
            {
                son[u] = v;
            }
        }
    }
}

int id[N], W[N], ts;
int top[N];

void dfs2(int u, int anc)
{
    id[u] = ++ts;
    W[ts] = A[u];
    top[u] = anc;

    if (son[u] == 0) return;

    dfs2(son[u], anc);

    for (int i = H[u]; i; i = E[i].n)
    {
        int v = E[i].v;
        if (v != father[u] && v != son[u])
        {
            dfs2(v, v);
        }
    }
}

struct TreeNode
{
    int l, r;
    LL sum, add;
} tr[N * 4];

void pushup(int u)
{
    tr[u].sum = tr[u * 2].sum + tr[u * 2 + 1].sum;
}

void pushdown(int u)
{
    if (tr[u].add)
    {
        auto& ls = tr[u * 2], & rs = tr[u * 2 + 1];
        ls.sum += (ls.r - ls.l + 1) * tr[u].add;
        ls.add += tr[u].add;
        rs.sum += (rs.r - rs.l + 1) * tr[u].add;
        rs.add += tr[u].add;
        tr[u].add = 0;
    }
}

void build(int u, int l, int r)
{
    tr[u] = { l, r };
    if (l == r)
    {
        tr[u].sum = W[l];
        return;
    }

    int mid = (l + r) / 2;
    build(u * 2, l, mid), build(u * 2 + 1, mid + 1, r);
    pushup(u);
}

LL query(int u, int l, int r)
{

    if (l <= tr[u].l && r >= tr[u].r)
    {
        return tr[u].sum;
    }

    pushdown(u);
    LL res = 0;
    int mid = (tr[u].l + tr[u].r) / 2;
    if (l <= mid) res += query(u * 2, l, r);
    if (r > mid) res += query(u * 2 + 1, l, r);
    return res;
}

void update(int u, int l, int r, int d)
{
    if (l <= tr[u].l && r >= tr[u].r)
    {
        tr[u].add += d;
        tr[u].sum += LL(tr[u].r - tr[u].l + 1) * d;
        return;
    }

    pushdown(u);
    int mid = (tr[u].l + tr[u].r) / 2;
    if (l <= mid) update(u * 2, l, r, d);
    if (r > mid) update(u * 2 + 1, l, r, d);
    pushup(u);
}

void updatePath(int a, int b, int d)
{
    while (top[a] != top[b])
    {
        if (depth[top[a]] < depth[top[b]]) std::swap(a, b);
        update(1, id[top[a]], id[a], d);
        a = father[top[a]];
    }

    if (depth[a] < depth[b]) std::swap(a, b);
    update(1, id[b], id[a], d);
}

void updateTree(int u, int d)
{
    update(1, id[u], id[u] + size[u] - 1, d);
}

LL queryPath(int a, int b)
{
    LL res = 0;
    while (top[a] != top[b])
    {
        if (depth[top[a]] < depth[top[b]]) std::swap(a, b);
        res += query(1, id[top[a]], id[a]);
        a = father[top[a]];
    }

    if (depth[a] < depth[b]) std::swap(a, b);
    res += query(1, id[b], id[a]);
    return res;
}

LL queryTree(int u)
{
    return query(1, id[u], id[u] + size[u] - 1);
}

int main()
{
    std::ios::sync_with_stdio(false);
    cin.tie(nullptr), cout.tie(nullptr);

    cin >> n;
    for (int i = 1; i <= n; i++) cin >> A[i];
    for (int i = 0; i < n - 1; i++)
    {
        int a, b; cin >> a >> b;
        add(a, b), add(b, a);
    }
    cin >> m;

    dfs1(1, -1);
    dfs2(1, 1);
    build(1, 1, n);

    for (int i = 0; i < m; i++)
    {
        int op; cin >> op;
        if (op == 1)
        {
            int a, b, d; cin >> a >> b >> d;
            updatePath(a, b, d);
        }
        else if (op == 2)
        {
            int u, d; cin >> u >> d;
            updateTree(u, d);
        }
        else if (op == 3)
        {
            int a, b; cin >> a >> b;
            cout << queryPath(a, b) << '\n';
        }
        else
        {
            int u; cin >> u;
            cout << queryTree(u) << '\n';
        }
    }

    return 0;
}
```

```c++
{

    {

    int mid = (tr[u].l + tr[u].r) / 2;
```

```c++
#include <cstdio>

using LL = long long;
using std::cin, std::cout;

const int N = 1e5 + 5;

int n, m;
struct TreeNode
{
    int s[2], p, v;
    int rev, sum;
} tr[N];

void pushup(int x)
{
    tr[x].sum = tr[tr[x].s[0]].sum ^ tr[x].v ^ tr[tr[x].s[1]].sum;
}

void reverse(int x)
{
    std::swap(tr[x].s[0], tr[x].s[1]);
    tr[x].rev ^= 1;
}

void pushdown(int x)
{
    if (tr[x].rev)
    {
        reverse(tr[x].s[0]);
        reverse(tr[x].s[1]);
        tr[x].rev = 0;
    }
}

bool isRoot(int x)
{
    int p = tr[x].p;
    return tr[p].s[0] != x && tr[p].s[1] != x;
}

void rotate(int x)
{
    int y = tr[x].p, z = tr[y].p;
    int k = tr[y].s[1] == x;

    if (!isRoot(y)) tr[z].s[tr[z].s[1] == y] = x;
    tr[x].p = z;
    tr[y].s[k] = tr[x].s[k ^ 1], tr[tr[x].s[k ^ 1]].p = y;
    tr[x].s[k ^ 1] = y, tr[y].p = x;

    pushup(y), pushup(x);
}

void splay(int x)
{
    static int stk[N];
    int top = 0;
    stk[++top] = x;
    for (int t = x; !isRoot(t); t = tr[t].p)
    {
        stk[++top] = tr[t].p;
    }
    while (top)
    {
        pushdown(stk[top--]);
    }

    while (!isRoot(x))
    {
        int y = tr[x].p, z = tr[y].p;
        if (!isRoot(y))
        {
            if ((tr[z].s[1] == y) ^ (tr[y].s[1] == x)) rotate(x);
            else rotate(y);
        }
        rotate(x);
    }
}

void access(int x)
{
    int z = x;
    for (int y = 0; x; y = x, x = tr[x].p)
    {
        splay(x);
        tr[x].s[1] = y;
        pushup(x);
    }
    splay(z);
}

void makeRoot(int x)
{
    access(x);
    reverse(x);
}

int findRoot(int x)
{
    access(x);
    while (tr[x].s[0])
    {
        pushdown(x);
        x = tr[x].s[0];
    }
    splay(x);
    return x;
}

void split(int x, int y)
{
    makeRoot(x);
    access(y);
}

void link(int x, int y)  
{
    makeRoot(x);
    if (findRoot(y) != x) tr[x].p = y;
}

void cut(int x, int y) 
{
    split(x, y);
    if (tr[y].s[0] == x && tr[x].s[1] == 0)
    {
        tr[x].p = tr[y].s[0] = 0;
        pushup(y);
    }
}
```

```c++
#include <cstring>

const int N = 5e4 + 5, M = 1e5 + 5, INF = M;

int n, m;
struct Edge
{
    int x, y, a, b;
} edge[M];

int P[N];

int find(int x)
{
    if (x != P[x]) P[x] = find(P[x]);
    return P[x];
}

struct TreeNode
{
    int s[2], p, v;
    int rev;
    int mx;
} tr[N + M];

void pushup(int x)
{
    tr[x].mx = x;  // 这一步至关重要
    for (int i = 0; i < 2; i++)
    {
        int mx = tr[tr[x].s[i]].mx;
        if (tr[mx].v > tr[tr[x].mx].v)
        {
            tr[x].mx = mx;
        }
    }
}

void reverse(int x)
{
    std::swap(tr[x].s[0], tr[x].s[1]);
    tr[x].rev ^= 1;
}

void pushdown(int x)
{
    if (tr[x].rev)
    {
        reverse(tr[x].s[0]), reverse(tr[x].s[1]);
        tr[x].rev = 0;
    }
}

bool isRoot(int x)
{
    int p = tr[x].p;
    return tr[p].s[0] != x && tr[p].s[1] != x;
}

void rotate(int x)
{
    int y = tr[x].p, z = tr[y].p;
    int k = tr[y].s[1] == x;

    if (!isRoot(y)) tr[z].s[tr[z].s[1] == y] = x;
    tr[x].p = z;
    tr[y].s[k] = tr[x].s[k ^ 1], tr[tr[x].s[k ^ 1]].p = y;
    tr[x].s[k ^ 1] = y, tr[y].p = x;
    pushup(y), pushup(x);
}

void splay(int x)
{
    static int stk[N + M];
    int top = 0;
    int cur = x;
    stk[++top] = cur;
    while (!isRoot(cur))
    {
        cur = tr[cur].p;
        stk[++top] = cur;
    }
    while (top) pushdown(stk[top--]);
    
    while (!isRoot(x))
    {
        int y = tr[x].p, z = tr[y].p;
        if (!isRoot(y))
        {
            if ((tr[y].s[1] == x) ^ (tr[z].s[1] == y)) rotate(x);
            else rotate(y);
        }
        rotate(x);
    }
}

void access(int x)
{
    int z = x;
    for (int y = 0; x; y = x, x = tr[x].p)
    {
        splay(x);
        tr[x].s[1] = y;
        pushup(x);
    }
    splay(z);
}

void makeRoot(int x)
{
    access(x);
    reverse(x);
}

int findRoot(int x)
{
    access(x);
    while (tr[x].s[0]) x = tr[x].s[0];
    splay(x);
    return x;
}

void split(int x, int y)
{
    makeRoot(x);
    access(y);
}

void link(int x, int y)
{
    makeRoot(x);
    if (findRoot(y) != x) tr[x].p = y;
}

{
    split(x, y);
    if (tr[y].s[0] == x && tr[x].s[1] == 0)
    {
        tr[x].p = tr[y].s[0] = 0;
        pushup(y);
    }
}

{
    std::ios::sync_with_stdio(false);
    cin.tie(nullptr), cout.tie(nullptr);

    cin >> n >> m;
    for (int i = 1; i <= m; i++)
    {
        int x, y, a, b; cin >> x >> y >> a >> b;
        edge[i] = { x, y, a, b };
    }
    for (int i = 1; i <= n; i++) P[i] = i;
    std::sort(edge + 1, edge + 1 + m,
        [](const Edge& u, const Edge& v)
        {
            return u.a < v.a;
        });
    for (int i = 1; i <= n + m; i++)
    {
        if (i > n) tr[i].v = edge[i - n].b;
        tr[i].mx = i;
    }

    int res = INF;
    for (int i = 1; i <= m; i++)
    {
        auto [x, y, a, b] = edge[i];
        if (find(x) == find(y))
        {
            split(x, y);
            int t = tr[y].mx;
            if (b < tr[t].v)
            {
                cut(edge[t - n].x, t), cut(t, edge[t - n].y);
                link(x, i + n), link(i + n, y);
            }
        }
        else
        {
            link(x, i + n), link(i + n, y);
            P[find(x)] = find(y);
        }

        if (find(1) == find(n))
        {
            split(n, 1);
            res = std::min(res, a + tr[tr[1].mx].v);
        }
    }

    if (res == INF) res = -1;
    cout << res << '\n';

    return 0;
}
```

```c++
#include <cstdio>
#include <cstring>
#include <iostream>
#include <algorithm>

using LL = long long;
using std::cin, std::cout;

const int N = 510, M = 5000 + N;

int n, m;
int L[M], R[M], up[M], dn[M], sum[N], row[M], col[M];
int idx;

void init()
{
    for (int i = 0; i <= m; i++)
    {
        L[i] = i - 1, R[i] = i + 1;
        up[i] = dn[i] = i;
    }
    L[0] = m, R[m] = 0;
    idx = m;
}

void add(int hh, int& tt, int x, int y)
{
    ++idx;
    row[idx] = x, col[idx] = y, sum[y]++;
    up[idx] = y, dn[idx] = dn[y], up[dn[y]] = idx, dn[y] = idx;
    L[idx] = hh, R[idx] = tt, R[hh] = L[tt] = idx;
    tt = idx;
}

void remove(int p)
{
    R[L[p]] = R[p], L[R[p]] = L[p];
    for (int i = dn[p]; i != p; i = dn[i])
    {
        for (int j = R[i]; j != i; j = R[j])
        {
            sum[col[j]]--;
            up[dn[j]] = up[j], dn[up[j]] = dn[j];
        }
    }
}

void resume(int p)
{
    for (int i = up[p]; i != p; i = up[i])
    {
        for (int j = L[i]; j != i; j = L[j])
        {
            up[dn[j]] = dn[up[j]] = j;
            sum[col[j]]++;
        }
    }
    R[L[p]] = L[R[p]] = p;
}

int res[N], top;

bool dfs()
{
    if (R[0] == 0) return true;

    int p = R[0];
    for (int i = R[0]; i != 0; i = R[i])
        if (sum[i] < sum[p])
            p = i;

    remove(p);

    for (int i = dn[p]; i != p; i = dn[i])
    {
        res[++top] = row[i];
        for (int j = R[i]; j != i; j = R[j]) remove(col[j]);
        if (dfs()) return true;
        for (int j = L[i]; j != i; j = L[j]) resume(col[j]);
        top--;
    }

    resume(p);

    return false;
}

int main()
{
    std::ios::sync_with_stdio(false);
    cin.tie(nullptr), cout.tie(nullptr);

    cin >> n >> m;
    init();

    for (int i = 1; i <= n; i++)
    {
        int hh = idx + 1, tt = idx + 1;
        for (int j = 1; j <= m; j++)
        {
            int a; cin >> a;
            if (a) add(hh, tt, i, j);
        }
    }

    if (dfs())
    {
        for (int i = 1; i <= top; i++) cout << res[i] << ' ';
        cout << '\n';
    }
    else cout << "No Solution!" << '\n';

    return 0;
}
```

#### 重复覆盖问题

题目：https://www.acwing.com/problem/content/2715/

题解：https://www.acwing.com/solution/content/160705/

```c++
#include <cstdio>
#include <cstring>
#include <iostream>
#include <algorithm>

using LL = long long;
using std::cin, std::cout;

const int N = 1e4 + 110, M = 110;

int n, m;
int L[N], R[N], up[N], dn[N], S[M], row[N], col[N];
int idx;
int res[M];

void init()
{
    for (int i = 0; i <= m; i++)
    {
        L[i] = i - 1, R[i] = i + 1;
        up[i] = dn[i] = i;
    }
    L[0] = m, R[m] = 0;
    idx = m;
}

void add(int& hh, int& tt, int x, int y)
{
    ++idx;
    row[idx] = x, col[idx] = y, S[y]++;
    L[idx] = hh, R[idx] = tt;
    up[idx] = y, dn[idx] = dn[y];
    R[hh] = L[tt] = idx;
    up[dn[y]] = idx, dn[y] = idx;
    tt = idx;
}

void remove(int p)  // p不是列的编号，而是十字链表节点的编号
{
    for (int i = dn[p]; i != p; i = dn[i])
    {
        L[R[i]] = L[i];
        R[L[i]] = R[i];
    }
}

void resume(int p)  // 和remove(p)同理
{
    for (int i = up[p]; i != p; i = up[i])
    {
        L[R[i]] = R[L[i]] = i;
    }
}

bool st[M];

int h()  // 估价函数
{
    memset(st, 0, sizeof st);
    int cnt = 0;
    for (int i = R[0]; i != 0; i = R[i])
        if (!st[i])
        {
            cnt++;
            st[i] = true;
            for (int j = dn[i]; j != i; j = dn[j])
                for (int k = R[j]; k != j; k = R[k])
                {
                    st[col[k]] = true;
                }
        }
    return cnt;
}

bool dfs(int k, int depth)
{
    if (k + h() > depth) return false;
    if (!R[0]) return true;

    int p = R[0];
    for (int i = R[0]; i != 0; i = R[i])
        if (S[i] < S[p])
            p = i;

    for (int i = dn[p]; i != p; i = dn[i])
    {
        res[k] = row[i];
        remove(i);
        for (int j = R[i]; j != i; j = R[j]) remove(j);
        if (dfs(k + 1, depth)) return true;
        // 以下两行顺序不能互换
        for (int j = L[i]; j != i; j = L[j]) resume(j);
        resume(i);
    }

    return false;
}

int main()
{
    std::ios::sync_with_stdio(false);
    cin.tie(nullptr), cout.tie(nullptr);

    cin >> n >> m;
    init();
    for (int i = 1; i <= n; i++)
    {
        int hh = idx + 1, tt = idx + 1;
        for (int j = 1; j <= m; j++)
        {
            int a; cin >> a;
            if (a) add(hh, tt, i, j);
        }
    }

    int depth = 1;
    while (!dfs(0, depth)) depth++;

    cout << depth << '\n';
    for (int i = 0; i < depth; i++) cout << res[i] << ' ';
    cout << '\n';

    return 0;
}
```

#### 破坏正方形

```c++
#include <algorithm>
#include <vector>

    for (int i = up[p]; i != p; i = up[i])
    {
        L[R[i]] = R[L[i]] = i;
    }
}

{
    if (k + h() > depth) return false;
    if (!R[0]) return true;

    int p = R[0];
    for (int i = R[0]; i != 0; i = R[i])
        if (S[i] < S[p])
            p = i;

    for (int i = dn[p]; i != p; i = dn[i])
    {
        remove(i);
        for (int j = R[i]; j != i; j = R[j]) remove(j);
        if (dfs(k + 1, depth)) return true;
        for (int j = L[i]; j != i; j = L[j]) resume(j);
        resume(i);
    }

    return false;
}

void solve()
{
    m = 0;
    memset(st, 0, sizeof st);
    for (int i = 1; i < N; i++)
    {
        square[i].clear();
        match[i].clear();
    }

    cin >> n;
    int t; cin >> t;
    while (t--)
    {
        int x; cin >> x;
        st[x] = true;
    }
    
    
    for (int i = 1; i <= n; i++)
    {
        for (int a = 1; a + i - 1 <= n; a++)
            for (int b = 1; b + i - 1 <= n; b++)
            {
                ++m;
                for (int j = 1; j <= i; j++)
                {
                    square[m].push_back((a - 1) * n + (a - 1) * (n + 1) + b + j - 1);
                    square[m].push_back(a * n + (a - 1) * (n + 1) + b + (j - 1) * (n * 2 + 1));
                    square[m].push_back((a + i - 1) * n + (a + i - 1) * (n + 1) + b + j - 1);
                    square[m].push_back(a * n + (a - 1) * (n + 1) + b + i + (j - 1) *  (n * 2 + 1));
                }

                bool rb = false;
                for (int x : square[m])
                    if (st[x])
                    {
                        square[m--].clear();
                        rb = true;
                        break;
                    }

                if (!rb)
                {
                    for (int x : square[m])
                    {
                        match[x].push_back(m);
                    }
                }
            }
    }

    init();  // 初始化十字链表

    for (int i = 1; i < N; i++)
    {
        int hh = idx + 1, tt = idx + 1;
        for (int x : match[i])
        {
            add(hh, tt, i, x);
        }
    }

    int depth = 0;
    while (!dfs(0, depth)) depth++;
    cout << depth << '\n';
}

int main()
{
    std::ios::sync_with_stdio(false);
    cin.tie(nullptr), cout.tie(nullptr);

    int t; cin >> t;
    while (t--)
        solve();

    return 0;
}
```

```c++
}
```

## 左偏树

### 左偏树

题目：https://www.acwing.com/problem/content/description/2716/

题解：https://www.acwing.com/solution/content/161640/

```c++
#include <cstdio>
#include <cstring>
#include <iostream>
#include <algorithm>

using LL = long long;
using std::cin, std::cout;

const int N = 2e5 + 5, INF = 1e9 + 10;

int n;
struct TreeNode
{
    int v, d, l, r;
} tr[N];
int idx;

bool less(int x, int y)
{
    if (tr[x].v != tr[y].v) return tr[x].v < tr[y].v;
    return x < y;
}

int merge(int x, int y)
{
    if (!x || !y) return x + y;

    if (less(y, x)) std::swap(x, y);
    tr[x].r = merge(tr[x].r, y);
    if (tr[tr[x].l].d < tr[tr[x].r].d) std::swap(tr[x].l, tr[x].r);
    tr[x].d = tr[tr[x].r].d + 1;
    return x;
}

int P[N];

int find(int x)
{
    if (x != P[x]) P[x] = find(P[x]);
    return P[x];
}

int main()
{
    std::ios::sync_with_stdio(false);
    cin.tie(nullptr), cout.tie(nullptr);

    // tr[0].v = INF;  // 保证删除的时候，左右子树有一个为空的情况正确。在这个代码下这行没有必要。

    cin >> n;
    while (n--)
    {
        int op, x; cin >> op >> x;
        if (op == 1)
        {
            tr[++idx] = { x, 1 };
            P[idx] = idx;
        }
        else if (op == 2)
        {
            int y;  cin >> y;
            x = find(x), y = find(y);
            if (x != y)
            {
                int z = merge(x, y);
                P[x] = P[y] = z;
            }
        }
        else if (op == 3)
        {
            cout << tr[find(x)].v << '\n';
        }
        else
        {
            x = find(x);
            int y = merge(tr[x].l, tr[x].r);
            P[x] = P[y] = y;
        }
    }

    return 0;
}
```

### 数字序列

题目：https://www.acwing.com/problem/content/description/2727/

题解：https://www.luogu.com.cn/article/d4x7qil2

```c++
#include <cstdio>
#include <cstring>
#include <iostream>
#include <algorithm>

using LL = long long;
using std::cin, std::cout;

const int N = 1e6 + 5;

int n;
int B[N];

struct TreeNode
{
    int v, d, l, r;
} tr[N];

int merge(int x, int y)
{
    if (!x || !y) return x + y;

    if (tr[x].v < tr[y].v) std::swap(x, y);
    tr[x].r = merge(tr[x].r, y);
    if (tr[tr[x].l].d < tr[tr[x].r].d) std::swap(tr[x].l, tr[x].r);
    tr[x].d = tr[tr[x].r].d + 1;
    return x;
}

struct Range
{
    int root, end, cnt;
} stk[N];
int top;

int main()
{
    std::ios::sync_with_stdio(false);
    cin.tie(nullptr), cout.tie(nullptr);

    cin >> n;
    for (int i = 1; i <= n; i++)
    {
        int a; cin >> a;
        tr[i].v = a - i;
        tr[i].d = 1;
    }

    for (int i = 1; i <= n; i++)
    {
        Range cur = { i, i, 1 };
        while (top && tr[cur.root].v < tr[stk[top].root].v)
        {
            cur.root = merge(cur.root, stk[top].root);
            if (cur.cnt % 2 && stk[top].cnt % 2)
            {
                cur.root = merge(tr[cur.root].l, tr[cur.root].r);
            }
            cur.cnt += stk[top].cnt;
            --top;
        }
        stk[++top] = cur;
    }

    for (int i = 1, j = 1; i <= top; i++)
    {
        while (j <= stk[i].end) B[j++] = tr[stk[i].root].v;
    }

    LL res = 0;
    for (int i = 1; i <= n; i++) res += std::abs(tr[i].v - B[i]);
    cout << res << '\n';
    for (int i = 1; i <= n; i++) cout << B[i] + i << ' ';
    cout << '\n';

    return 0;
}
```

### K-单调

题目：https://www.acwing.com/problem/content/description/2723/

题解：https://www.acwing.com/solution/content/162011/

```c++
#include <cstdio>
#include <cstring>
#include <iostream>
#include <algorithm>

using LL = long long;
using std::cin, std::cout;

const int N = 1010;

int n, K;
int A[N];
int cost[N][N], f[N][15];

struct TreeNode
{
    int v, d, l, r;
} tr[N];

int merge(int x, int y)
{
    if (!x || !y) return x + y;

    if (tr[x].v < tr[y].v) std::swap(x, y);
    tr[x].r = merge(tr[x].r, y);
    if (tr[tr[x].l].d < tr[tr[x].r].d) std::swap(tr[x].l, tr[x].r);
    tr[x].d = tr[tr[x].r].d + 1;
    return x;
}

struct Range  // 维护递减区间
{
    int root;
    int cnt, sum;
    int tcnt, tsum;

    int cost()
    {
        int mid = tr[root].v;
        return mid * tcnt - tsum + sum - tsum - mid * (cnt - tcnt);
    }
} stk[N];

void getCost(int u)  // 以u为起点计算cost(u, ...)
{
    for (int i = u; i <= n; i++)
    {
        tr[i].d = 1;
        tr[i].l = tr[i].r = 0;
    }

    int top = 0, acc = 0;
    for (int i = u; i <= n; i++)
    {
        Range cur{ i, 1, tr[i].v, 1, tr[i].v };
        while (top && tr[cur.root].v < tr[stk[top].root].v)
        {
            auto& pre = stk[top];
            acc -= pre.cost();
            bool isPop = cur.cnt % 2 && pre.cnt % 2;
            cur.root = merge(cur.root, pre.root);
            cur.cnt += pre.cnt, cur.sum += pre.sum;
            cur.tcnt += pre.tcnt, cur.tsum += pre.tsum;
            if (isPop)
            {
                cur.tcnt--, cur.tsum -= tr[cur.root].v;
                cur.root = merge(tr[cur.root].l, tr[cur.root].r);
            }
            top--;
        }
        stk[++top] = cur;
        acc += cur.cost();
        cost[u][i] = std::min(cost[u][i], acc);
    }
}

int main()
{
    std::ios::sync_with_stdio(false);
    cin.tie(nullptr), cout.tie(nullptr);

    cin >> n >> K;
    for (int i = 1; i <= n; i++) cin >> A[i];

    memset(cost, 0x3f, sizeof cost);
    for (int i = 1; i <= n; i++) tr[i].v = A[i] - i;
    for (int i = 1; i <= n; i++) getCost(i);
    for (int i = 1; i <= n; i++) tr[i].v = -A[i] - i;
    for (int i = 1; i <= n; i++) getCost(i);

    memset(f, 0x3f, sizeof f);
    f[0][0] = 0;
    for (int i = 1; i <= n; i++)
        for (int j = 1; j <= std::min(i, K); j++)
            for (int k = 1; k <= i - j + 1; k++)
                f[i][j] = std::min(f[i][j], f[i - k][j - 1] + cost[i - k + 1][i]);

    cout << f[n][K] << '\n';

    return 0;
}
```

## 后缀数组

### 后缀数组

题目：https://www.acwing.com/problem/content/description/2717/

题解：https://www.acwing.com/solution/content/163123/

```c++
#include <cstdio>
#include <cstring>
#include <iostream>
#include <algorithm>

using LL = long long;
using std::cin, std::cout;

const int N = 1e6 + 5;

int n, m;
std::string S;
int sa[N], H[N];
int X[N], Y[N], rk[N], cnt[N];

void getSA()
{
    for (int i = 1; i <= n; i++) cnt[X[i] = S[i]]++;
    for (int i = 2; i <= m; i++) cnt[i] += cnt[i - 1];
    for (int i = n; i >= 1; i--) sa[cnt[X[i]]--] = i;

    for (int k = 1; k <= n; k *= 2)
    {
        int p = 0;
        for (int i = n - k + 1; i <= n; i++) Y[++p] = i;
        for (int i = 1; i <= n; i++)
            if (sa[i] > k)
                Y[++p] = sa[i] - k;

        for (int i = 1; i <= m; i++) cnt[i] = 0;
        for (int i = 1; i <= n; i++) cnt[X[i]]++;
        for (int i = 2; i <= m; i++) cnt[i] += cnt[i - 1];
        for (int i = n; i >= 1; i--) sa[cnt[X[Y[i]]]--] = Y[i];

        std::swap(X, Y);
        p = X[sa[1]] = 1;
        for (int i = 2; i <= n; i++)
        {
            X[sa[i]] = (Y[sa[i]] == Y[sa[i - 1]]) && (Y[sa[i] + k] == Y[sa[i - 1] + k]) ? p : ++p;
        }

        if (p == n) break;
        m = p;
    }
}

void getH()
{
    for (int i = 1; i <= n; i++) rk[sa[i]] = i;

    int k = 0;
    for (int i = 1; i <= n; i++)
    {
        if (rk[i] == 1) continue;
        if (k) k--;

        int j = sa[rk[i] - 1];
        while (i + k <= n && j + k <= n && S[i + k] == S[j + k]) k++;
        H[rk[i]] = k;
    }
}

int main()
{
    std::ios::sync_with_stdio(false);
    cin.tie(nullptr), cout.tie(nullptr);

    cin >> S;
    n = S.size(), m = 122;
    S = ' ' + S;

    getSA();
    getH();

    for (int i = 1; i <= n; i++) cout << sa[i] << ' ';
    cout << '\n';
    for (int i = 1; i <= n; i++) cout << H[i] << ' ';
    cout << '\n';

    return 0;
}
```

```c++
//const LL INF = 1e18 + 10;  // 由于浮点数的特性，1e18+10会变成1e18，会产生严重的错误
const LL INF = LL(1e18) + 10;

int n, m;
std::string S;
int A[N];

int sa[N], H[N];
int X[N], Y[N], rk[N], cnt[N];

void getSA()
{
    for (int i = 1; i <= n; i++) cnt[X[i] = S[i]]++;
    std::partial_sum(cnt + 1, cnt + 1 + m, cnt + 1);
    for (int i = n; i >= 1; i--) sa[cnt[X[i]]--] = i;

    for (int k = 1; k <= n; k *= 2)
    {
        int p = 0;
        for (int i = n - k + 1; i <= n; i++) Y[++p] = i;
        for (int i = 1; i <= n; i++)
            if (sa[i] > k)
                Y[++p] = sa[i] - k;

        std::fill_n(cnt + 1, m, 0);
        for (int i = 1; i <= n; i++) cnt[X[i]]++;
        std::partial_sum(cnt + 1, cnt + 1 + m, cnt + 1);
        for (int i = n; i >= 1; i--) sa[cnt[X[Y[i]]]--] = Y[i];

        std::swap(X, Y);
        p = X[sa[1]] = 1;
        for (int i = 2; i <= n; i++)
        {
            X[sa[i]] = (Y[sa[i]] == Y[sa[i - 1]]) && (Y[sa[i] + k] == Y[sa[i - 1] + k]) ? p : ++p;
        }
        if (p == n) break;
        m = p;
    }
}

void getH()
{
    for (int i = 1; i <= n; i++) rk[sa[i]] = i;

    int k = 0;
    for (int i = 1; i <= n; i++)
    {
        if (rk[i] == 1) continue;
        
        if (k) k--;
        int j = sa[rk[i] - 1];
        while (i + k <= n && j + k <= n && S[i + k] == S[j + k]) k++;
        H[rk[i]] = k;
    }
}

struct UFS
{
    int p, cnt;
    LL mx1, mx2, mn1, mn2;
} U[N];

int find(int x)
{
    if (U[x].p != x) U[x].p = find(U[x].p);
    return U[x].p;
}

std::pair<LL, LL> res[N];
std::vector<int> hs[N];

LL get(int a)  // 计算C(a, 2)
{
    return LL(a) * (a - 1) / 2;
}

std::pair<LL, LL> calc(int r)
{
    static LL tot = 0, mx = -INF;

    for (int rank : hs[r])
    {
        int a = find(rank - 1), b = find(rank);
        tot -= get(U[a].cnt) + get(U[b].cnt);
        U[a].p = b;
        U[b].cnt += U[a].cnt;
        tot += get(U[b].cnt);

        if (U[a].mx1 >= U[b].mx1)
        {
            U[b].mx2 = std::max(U[b].mx1, U[a].mx2);
            U[b].mx1 = U[a].mx1;
        }
        else if (U[a].mx1 > U[b].mx2)
        {
            U[b].mx2 = U[a].mx1;
        }
        if (U[a].mn1 <= U[b].mn1)
        {
            U[b].mn2 = std::min(U[b].mn1, U[a].mn2);
            U[b].mn1 = U[a].mn1;
        }
        else if (U[a].mn1 < U[b].mn2)
        {
            U[b].mn2 = U[a].mn1;
        }
        mx = std::max({mx, U[b].mx1 * U[b].mx2, U[b].mn1 * U[b].mn2});
    }

    return { tot, mx == -INF ? 0 : mx };
}

int main()
{

    cin.tie(nullptr), cout.tie(nullptr);

    cin >> n >> S;

    getSA();
    getH();

    for (int i = 1; i <= n; i++)  // 初始化并查集
    {
        U[i] = { i, 1, A[sa[i]], -INF, A[sa[i]], INF };
    }
    for (int i = 2; i <= n; i++) hs[H[i]].push_back(i);
    for (int i = n - 1; i >= 0; i--) res[i] = calc(i);

    for (int i = 0; i <= n - 1; i++)
    {
        auto [a, b] = res[i];
        cout << a << ' ' << b << '\n';
    }

    return 0;
}
```

```c++
    }

}
```

## 后缀自动机

### 后缀自动机

题目：https://www.acwing.com/problem/content/2768/

```c++
#include <cstdio>
#include <iostream>
#include <algorithm>
#include <numeric>

using LL = long long;
using std::cin, std::cout;

const int N = int(2e6) + 5;

std::string str;
LL f[N], res;

struct
{
    int len, fa;
    int ch[26];
} S[N];
int tot = 1, last = 1; // 1代表endpos(空集)

void extend(int c)
{
    int p = last, np = ++tot;  // 分配节点
    // 初始化新的节点
    S[np].len = S[p].len + 1;
    for (; p && !S[p].ch[c]; p = S[p].fa) S[p].ch[c] = np;

    f[tot] = 1;

    if (!p) S[np].fa = 1;
    else
    {
        int q = S[p].ch[c];
        if (S[q].len == S[p].len + 1)
        {
            S[np].fa = q;
        }
        else
        {
            int nq = ++tot;
            S[nq] = S[q], S[nq].len = S[p].len + 1;

            S[np].fa = S[q].fa = nq;
            for (; p && S[p].ch[c] == q; p = S[p].fa) S[p].ch[c] = nq;
        }
    }
    last = np;
}

int H[N];
struct
{
    int v, n;
} E[N];
int idx;

void add(int a, int b)
{
    E[++idx] = { b, H[a] }, H[a] = idx;
}

void dfs(int u)
{
    for (int i = H[u]; i; i = E[i].n)
    {
        int v = E[i].v;
        dfs(v);
        f[u] += f[v];
    }
    if (f[u] > 1) res = std::max(res, f[u] * S[u].len);
}

int main()
{
    std::ios::sync_with_stdio(false);
    cin.tie(nullptr), cout.tie(nullptr);

    cin >> str;
    for (char c : str) extend(c - 'a');

    for (int i = 2; i <= tot; i++) add(S[i].fa, i);

    dfs(1);

    cout << res << '\n';

    return 0;
}
```

### 树

题目：https://www.acwing.com/problem/content/description/254/

题解：https://www.acwing.com/solution/content/164945/

```c++
#include <cstdio>
#include <iostream>
#include <algorithm>
#include <numeric>
#include <vector>

using LL = long long;
using std::cin, std::cout;

const int N = int(1e4) + 5;

int n, m;
int H[N];
struct
{
    int v, w, n;
} E[N * 2];
int idx;

void add(int a, int b, int w)
{
    E[++idx] = { b, w, H[a] }, H[a] = idx;
}

bool st[N];

int getSize(int u, int fa)
{
    if (st[u]) return 0;

    int sum = 1;
    for (int i = H[u]; i; i = E[i].n)
        if (int v = E[i].v; v != fa)
        {
            sum += getSize(v, u);
        }
    return sum;
}

int getWC(int u, int fa, int size, int& wc)
{
    if (st[u]) return 0;

    int sum = 1, mx = 0;
    for (int i = H[u]; i; i = E[i].n)
        if (int v = E[i].v; v != fa)
        {
            int t = getWC(v, u, size, wc);
            mx = std::max(mx, t);
            sum += t;
        }
    mx = std::max(mx, size - sum);
    if (mx <= size / 2) wc = u;
    return sum;
}


void getDist(int u, int fa, int cur, std::vector<int>& vec)
{
    if (st[u]) return;

    vec.push_back(cur);
    for (int i = H[u]; i; i = E[i].n)
        if (int v = E[i].v, w = E[i].w; v != fa)
        {
            getDist(v, u, cur + w, vec);
        }
}

int get(std::vector<int>& vec)
{
    int res = 0;
    std::sort(vec.begin(), vec.end());
    for (int i = vec.size() - 1, j = -1; i >= 0; i--)
    {
        while (j + 1 < i && vec[i] + vec[j + 1] <= m) j++;
        j = std::min(j, i - 1);
        res += j + 1;
    }
    return res;
}

std::vector<int> dist;

int dc(int u)
{
    if (st[u]) return 0;

    int size = getSize(u, -1);
    getWC(u, -1, size, u);
    st[u] = true;

    int res = 0;
    dist.clear();
    for (int i = H[u]; i; i = E[i].n)
    {
        int v = E[i].v, w = E[i].w;
        std::vector<int> vec;
        getDist(v, u, w, vec);
        res -= get(vec);
        dist.insert(dist.end(), vec.begin(), vec.end());
    }
    res += get(dist);

    for (int d : dist)
        if (d <= m) res++;

    for (int i = H[u]; i; i = E[i].n)
    {
        int v = E[i].v;
        res += dc(v);
    }
    return res;
}

void solve()
{
    std::fill_n(st, N, false);
    std::fill_n(H, N, 0);
    idx = 0;
    for (int i = 0; i < n - 1; i++)
    {
        int a, b, w; cin >> a >> b >> w;
        add(a, b, w), add(b, a, w);
    }

    cout << dc(0) << '\n';
}

int main()
{
    std::ios::sync_with_stdio(false);
    cin.tie(nullptr), cout.tie(nullptr);

    while (cin >> n >> m, n || m)
        solve();

    return 0;
}
```

### 权值

```c++

using LL = long long;
using std::cin, std::cout;

{
    int v, w, n;
} E[N * 2];
int idx;

void add(int a, int b, int w)
{

int getSize(int u, int fa)
{
    if (st[u]) return 0;

    int sum = 1;
    for (int i = H[u]; i; i = E[i].n)
        if (int v = E[i].v; v != fa)
        {
            sum += getSize(v, u);
        }
    return sum;
}

{
    if (st[u]) return 0;

    int sum = 1, mx = 0;
    for (int i = H[u]; i; i = E[i].n)
        if (int v = E[i].v; v != fa)
        {
            int t = getWC(v, u, size, wc);
            sum += t;
            mx = std::max(mx, t);
        }
    mx = std::max(mx, size - sum);
    if (mx <= size / 2) wc = u;
    return sum;

void getDist(int u, int fa, int dist, int cnt, std::vector<std::pair<int, int>>& vec)
{
    if (st[u] || dist > m) return;

    vec.push_back({ dist, cnt });

    for (int i = H[u]; i; i = E[i].n)
        if (int v = E[i].v, w = E[i].w; v != fa)
        {
            getDist(v, u, dist + w, cnt + 1, vec);
        }
}

int ht[M];

void dc(int u)
{
    if (st[u]) return;

    int size = getSize(u, -1);
    getWC(u, -1, size, u);
    st[u] = true;

    std::vector<std::pair<int, int>> p;
    for (int i = H[u]; i; i = E[i].n)
    {
        int v = E[i].v, w = E[i].w;
        std::vector<std::pair<int, int>> q;
        getDist(v, u, w, 1, q);
        for (auto [dist, cnt] : q)
        {
            if (dist == m) res = std::min(res, cnt);
            else res = std::min(res, cnt + ht[m - dist]);
        }
        for (auto [dist, cnt] : q) ht[dist] = std::min(ht[dist], cnt);
        p.insert(p.end(), q.begin(), q.end());
    }
    for (auto [dist, cnt] : p) ht[dist] = INF;  // 这一步至关重要，如果每次dc都std::fill会超时

    }
}

        add(a, b, w), add(b, a, w);
```
