## 搜索

### DFS

#### 排列数字

题目：[排列数字](https://www.acwing.com/problem/content/844/)

```c++
#include <cstdio>
#include <cstring>
#include <iostream>
#include <algorithm>

using LL = long long;
using std::cin, std::cout;

const int N = 10;

int n, A[N];
bool st[N];

void dfs(int i)
{
    if (i == n)
    {
        for (int j = 0; j < n; j++) cout << A[j] << ' ';
        cout << '\n';
        return;
    }

    for (int j = 1; j <= n; j++)
        if (!st[j])
        {
            st[j] = true;
            A[i] = j;
            dfs(i + 1);
            st[j] = false;
        }
}

int main()
{
    std::ios::sync_with_stdio(false);
    cin.tie(nullptr), cout.tie(nullptr);

    cin >> n;
    dfs(0);

    return 0;
}
```

#### n-皇后问题

题目：[n-皇后问题](https://www.acwing.com/problem/content/845/)

```c++
#include <cstdio>
#include <cstring>
#include <iostream>
#include <algorithm>
#include <bitset>

using LL = long long;
using std::cin, std::cout;

const int N = 10;

int n;
char board[N][N];
std::bitset<N> col;
std::bitset<N * 2> dg, bdg;

void dfs(int i)
{
    if (i == n)
    {
        for (int j = 0; j < n; j++) cout << board[j] << '\n';
        cout << '\n';
        return;
    }

    for (int j = 0; j < n; j++)
    {
        if (col[j] || dg[i + j] || bdg[j - i + n - 1]) continue;
        col[j] = dg[i + j] = bdg[j - i + n - 1] = true;
        board[i][j] = 'Q';
        dfs(i + 1);
        col[j] = dg[i + j] = bdg[j - i + n - 1] = false;
        board[i][j] = '.';
    }
}

int main()
{
    std::ios::sync_with_stdio(false);
    cin.tie(nullptr), cout.tie(nullptr);

    cin >> n;
    for (int i = 0; i < n; i++)
        for (int j = 0; j < n; j++)
            board[i][j] = '.';
    dfs(0);

    return 0;
}
```



### BFS

#### 走迷宫

题目：[走迷宫](https://www.acwing.com/problem/content/846/)

```c++
#include <cstdio>
#include <cstring>
#include <iostream>
#include <algorithm>

using LL = long long;
using std::cin, std::cout;

const int N = 110;

int n, m;
int G[N][N];

struct Point
{
    int x, y;
};

Point Q[N * N];
bool st[N][N];
int dist[N][N];
int dx[4] = {-1, 0, 1, 0};
int dy[4] = {0, 1, 0, -1};

int bfs()
{
    int hh = 0, tt = 0;
    Q[tt++] = {0, 0};

    while (hh < tt)
    {
        auto [x, y] = Q[hh++];

        for (int i = 0; i < 4; i++)
        {
            int nx = x + dx[i], ny = y + dy[i];
            if (nx < 0 || nx >= n || ny < 0 || ny >= m) continue;
            if (G[nx][ny] == 1 || st[nx][ny]) continue;
            st[nx][ny] = true;
            dist[nx][ny] = dist[x][y] + 1;
            Q[tt++] = {nx, ny};
        }
    }

    return dist[n - 1][m - 1];
}

int main()
{
    std::ios::sync_with_stdio(false);
    cin.tie(nullptr), cout.tie(nullptr);

    cin >> n >> m;
    for (int i = 0; i < n; i++)
        for (int j = 0; j < m; j++)
            cin >> G[i][j];

    cout << bfs() << '\n';
    return 0;
}
```

#### 八数码

题目：[八数码](https://www.acwing.com/problem/content/description/847/)

```c++
#include <cstdio>
#include <cstring>
#include <iostream>
#include <algorithm>
#include <queue>
#include <unordered_map>

using LL = long long;
using std::cin, std::cout;
using std::string;

string start(9, 0), end = "12345678x";
std::unordered_map<string, int> dist;
std::queue<string> Q;

int dx[4] = {-1, 0, 1, 0};
int dy[4] = {0, 1, 0, -1};

int bfs()
{
    if (start == end) return 0;

    Q.push(start);
    dist[start] = 0;

    while (!Q.empty())
    {
        string s = Q.front(); Q.pop();
        int pre = s.find('x'), d = dist[s];
        int px = pre / 3, py = pre % 3;
        for (int i = 0; i < 4; i++)
        {
            int x = px + dx[i], y = py + dy[i];
            if (x >= 0 && x < 3 && y >= 0 && y < 3)
            {
                int cur = x * 3 + y;
                std::swap(s[pre], s[cur]);
                if (!dist.count(s))
                {
                    if (s == end) return d + 1;
                    dist[s] = d + 1;
                    Q.push(s);
                }
                std::swap(s[pre], s[cur]);
            }
        }
    }

    return -1;
}

int main()
{
    std::ios::sync_with_stdio(false);
    cin.tie(nullptr), cout.tie(nullptr);

    for (int i = 0; i < 9; i++) cin >> start[i];
    cout << bfs() << '\n';

    return 0;
}
```



## 图的遍历

### 树与图的深度优先遍历

时间复杂度为$O(n+m)$，$n$表示点数，$m$表示边数

#### 树的重心

题目：[树的重心](https://www.acwing.com/problem/content/848/)

```c++
#include <cstdio>
#include <cstring>
#include <iostream>
#include <algorithm>

using LL = long long;
using std::cin, std::cout;

const int N = 1e5 + 5;

int n;
int H[N];
struct EdgeNode
{
    int v, n;
} E[N * 2];
int idx;

void add(int a, int b)
{
    E[++idx] = {b, H[a]}, H[a] = idx;
}

int res;

int dfs(int u, int fa)
{
    int cnt = 1, mx = 0;

    for (int p = H[u]; p; p = E[p].n)
        if (int v = E[p].v; v != fa)
        {
            int t = dfs(v, u);
            cnt += t;
            mx = std::max(mx, t);
        }

    mx = std::max(mx, n - cnt);
    res = std::min(res, mx);
    
    return cnt;
}

int main()
{
    std::ios::sync_with_stdio(false);
    cin.tie(nullptr), cout.tie(nullptr);

    cin >> n;
    for (int i = 1; i < n; i++)
    {
        int a, b; cin >> a >> b;
        add(a, b), add(b, a);
    }

    res = N;
    dfs(1, -1);
    cout << res << '\n';

    return 0;
}
```

### 树与图的广度优先遍历

时间复杂度为$O(n+m)$，$n$表示点数，$m$表示边数

#### 图中点的层次

题目：[图中点的层次](https://www.acwing.com/problem/content/849/)

```c++
#include <cstdio>
#include <cstring>
#include <iostream>
#include <algorithm>

using LL = long long;
using std::cin, std::cout;

const int N = 1e5 + 5;

int n, m;
int H[N];
struct EdgeNode
{
    int v, n;
} E[N];
int idx;

void add(int a, int b)
{
    E[++idx] = {b, H[a]}, H[a] = idx;
}

int Q[N], hh, tt;
bool st[N];

{

}
```

## 拓扑排序

时间复杂度为$O(n+m)$，$n$表示点数，$m$表示边数

### 有向图的拓扑序列

题目：[有向图的拓扑序列](https://www.acwing.com/problem/content/850/)

```c++
#include <cstdio>
#include <cstring>
#include <iostream>
#include <algorithm>

using LL = long long;
using std::cin, std::cout;

const int N = 1e5 + 5;

int n, m;
int H[N];
struct EdgeNode
{
    int v, n;
} E[N];
int idx;

void add(int a, int b)
{
    E[++idx] = {b, H[a]}, H[a] = idx;
}

int indegree[N];
int Q[N], hh, tt;

bool topsort()
{
    for (int i = 1; i <= n; i++)
        if (indegree[i] == 0)
            Q[tt++] = i;

    while (hh < tt)
    {
        int u = Q[hh++];

        for (int p = H[u]; p; p = E[p].n)
        {
            int v = E[p].v;
            if (--indegree[v] == 0) Q[tt++] = v;
        }
    }

    return tt == n;
}

int main()
{
    std::ios::sync_with_stdio(false);
    cin.tie(nullptr), cout.tie(nullptr);

    cin >> n >> m;
    for (int i = 0; i < m; i++)
    {
        int a, b; cin >> a >> b;
        add(a, b);
        indegree[b]++;
    }

    if (topsort())
    {
        for (int i = 0; i < n; i++) cout << Q[i] << ' ';
        cout << '\n';
    }
    else cout << -1 << '\n';

    return 0;
}
```

## 最短路

### Dijkstra

#### 朴素Dijkstra

时间复杂度为$O(n^2+m)$，$n$表示点数，$m$表示边数

##### Dijkstra求最短路 I

题目：[Dijkstra求最短路 I](https://www.acwing.com/problem/content/851/)

题解：https://www.acwing.com/solution/content/38318/

```c++
#include <cstdio>
#include <cstring>
#include <iostream>
#include <algorithm>

using LL = long long;
using std::cin, std::cout;

const int N = 510, M = 1e5 + 5, INF = 0x3f3f3f3f;

int n, m;
int H[N];
struct EdgeNode
{
    int v, w, n;
} E[M];
int idx;

void add(int a, int b, int w)
{
    E[++idx] = {b, w, H[a]}, H[a] = idx;
}

bool st[N];
int dist[N];

void dijkstra()
{
    memset(dist, 0x3f, sizeof dist);
    dist[1] = 0;

    for (int i = 0; i < n; i++)
    {
        int u = -1;
        for (int j = 1; j <= n; j++)
            if (!st[j] && (u == -1 || dist[j] < dist[u]))
                u = j;

        st[u] = true;
        if (u == n) return;

        for (int p = H[u]; p; p = E[p].n)
        {
            int v = E[p].v, w = E[p].w;
            dist[v] = std::min(dist[v], dist[u] + w);
        }
    }
}

int main()
{
    std::ios::sync_with_stdio(false);
    cin.tie(nullptr), cout.tie(nullptr);

    cin >> n >> m;
    for (int i = 0; i < m; i++)
    {
        int a, b, w; cin >> a >> b >> w;
        add(a, b, w);
    }

    dijkstra();

    if (dist[n] != INF) cout << dist[n] << '\n';
    else cout << -1 << '\n';

    return 0;
}
```

#### 堆优化版Dijkstra

时间复杂度为$O(m \log_2 n)$，$n$表示点数，$m$表示边数

##### Dijkstra求最短路 II

题目：[Dijkstra求最短路 II](https://www.acwing.com/problem/content/852/)

```c++
#include <cstdio>
#include <cstring>
#include <iostream>
#include <algorithm>
#include <queue>

using LL = long long;
using std::cin, std::cout;

const int N = 15e4 + 5, INF = 0x3f3f3f3f;

int n, m;
int H[N];
struct EdgeNode
{
    int v, w, n;
} E[N];
int idx;

void add(int a, int b, int w)
{
    E[++idx] = {b, w, H[a]}, H[a] = idx;
}

bool st[N];
int dist[N];

void dijkstra()
{
    memset(dist, 0x3f, sizeof dist);
    dist[1] = 0;

    struct Node
    {
        int d, v;

        bool operator< (const Node &ct) const
        {
            return d >= ct.d;
        }
    };
    std::priority_queue<Node> Q;
    Q.push({0, 1});

    while (!Q.empty())
    {
        auto [d, u] = Q.top(); Q.pop();

        if (st[u]) continue;
        st[u] = true;
        if (u == n) return;
        
        for (int p = H[u]; p; p = E[p].n)
        {
            int v = E[p].v, w = E[p].w;
            if (d + w < dist[v])
            {
                dist[v] = d + w;
                Q.push({dist[v], v}); 
            }
        }
    }
}

int main()
{
    std::ios::sync_with_stdio(false);
    cin.tie(nullptr), cout.tie(nullptr);

    cin >> n >> m;
    for (int i = 0; i < m; i++)
    {
        int a, b, w; cin >> a >> b >> w;
        add(a, b, w);
    }

    dijkstra();

    if (dist[n] != INF) cout << dist[n] << '\n';
    else cout << -1 << '\n';

    return 0;
}
```

### bellman-ford

时间复杂度$O(nm)$，$n$表示点数，$m$表示边数

#### 有边数限制的最短路

题目：[有边数限制的最短路](https://www.acwing.com/problem/content/855/)

题解：https://www.acwing.com/solution/content/14088/

```c++
#include <cstdio>
#include <cstring>
#include <iostream>
#include <algorithm>

using LL = long long;
using std::cin, std::cout;

const int N = 510, M = 1e4 + 5, INF = 0x3f3f3f3f;

int n, m, K;
struct Edge
{
    int a, b, w;
} E[M];

int dist[N], back[N];

void bellmanFord()
{
    memset(dist, 0x3f, sizeof dist);
    dist[1] = 0;

    for (int i = 0; i < K; i++)
    {
        memcpy(back, dist, sizeof dist);
        for (int j = 0; j < m; j++)
        {
            auto [a, b, w] = E[j];
            dist[b] = std::min(dist[b], back[a] + w);
        }
    }
}

int main()
{
    std::ios::sync_with_stdio(false);
    cin.tie(nullptr), cout.tie(nullptr);

    cin >> n >> m >> K;
    for (int i = 0; i < m; i++) cin >> E[i].a >> E[i].b >> E[i].w;

    bellmanFord();

    if (dist[n] > INF / 2) cout << "impossible" << '\n';
    else cout << dist[n] << '\n';

    return 0;
}
```

1. 在迭代过程中，可能会更新`d[n] = min(d[n], last[i] + z)`，如果`last[i] = INF`和`z < 0`，则可能出现`d[n] < INF`的情况。但在理论上`d[n]`依然是正无穷。
1. bf算法在各方面都不如spfa，但题中的应用却只有bf算法才能胜任。

### spfa

时间复杂度平均情况下$O(m)$，最坏情况下$O(nm)$，$n$表示点数，$m$表示边数

#### spfa求最短路

**模板题：**

题目：[spfa求最短路](https://www.acwing.com/problem/content/853/)

题解：https://www.acwing.com/solution/content/105508/

```c++
#include <cstdio>
#include <cstring>
#include <iostream>
#include <algorithm>

using LL = long long;
using std::cin, std::cout;

const int N = 1e5 + 5, INF = 0x3f3f3f3f;

int n, m;
int H[N];
struct EdgeNode
{
    int v, w, n;
} E[N];
int idx;

void add(int a, int b, int w)
{
    E[++idx] = {b, w, H[a]}, H[a] = idx;
}

int dist[N];
int Q[N], hh, tt;
bool st[N];

void spfa()
{
    memset(dist, 0x3f, sizeof dist);
    dist[1] = 0;
    st[1] = true;
    Q[tt++] = 1;

    while (hh < tt)
    {
        int u = Q[hh++];
        st[u] = false;

        for (int p = H[u]; p; p = E[p].n)
        {
            int v = E[p].v, w = E[p].w;
            if (dist[u] + w < dist[v])
            {
                dist[v] = dist[u] + w;
                if (!st[v])
                {
                    st[v] = true;
                    Q[tt++] = v;
                }
            }
        }
    }
}

int main()
{
    std::ios::sync_with_stdio(false);
    cin.tie(nullptr), cout.tie(nullptr);

    cin >> n >> m;
    for (int i = 0; i < m; i++)
    {
        int a, b, w; cin >> a >> b >> w;
        add(a, b, w);
    }

    spfa();

    if (dist[n] != INF) cout << dist[n] << '\n';
    else cout << "impossible" << '\n';

    return 0;
}
```

#### spfa判断负环

**模板题：**

题目：[spfa判断负环](https://www.acwing.com/problem/content/854/)

```c++
#include <cstdio>
#include <cstring>
#include <iostream>
#include <algorithm>

using LL = long long;
using std::cin, std::cout;

const int N = 2010, M = 1e4 + 5, INF = 0x3f3f3f3f;

int n, m;
int H[N];
struct EdgeNode
{
    int v, w, n;
} E[M];
int idx;

void add(int a, int b, int w)
{
    E[++idx] = {b, w, H[a]}, H[a] = idx;
}

int dist[N], cnt[N];
int Q[N], hh, tt;
bool st[N];

bool spfa()
{
    for (int i = 1; i <= n; i++)
    {
        st[i] = true;
        Q[tt++] = i;
    }
    tt %= N;

    while (hh != tt)
    {
        int u = Q[hh++]; hh %= N;
        st[u] = false;

        for (int p = H[u]; p; p = E[p].n)
        {
            int v = E[p].v, w = E[p].w;
            if (dist[u] + w < dist[v])
            {
                dist[v] = dist[u] + w;
                cnt[v] = cnt[u] + 1;
                if (cnt[v] >= n) return true;
                if (!st[v])
                {
                    st[v] = true;
                    Q[tt++] = v; tt %= N;
                }
            }
        }
    }

    return false;
}

int main()
{
    std::ios::sync_with_stdio(false);
    cin.tie(nullptr), cout.tie(nullptr);

    cin >> n >> m;
    for (int i = 0; i < m; i++)
    {
        int a, b, w; cin >> a >> b >> w;
        add(a, b, w);
    }

    cout << (spfa() ? "Yes" : "No") << '\n';

    return 0;
}
```

### floyd

时间复杂度$O(n^3)$，$n$表示点数

#### Floyd求最短路

题目：[Floyd求最短路](https://www.acwing.com/problem/content/856/)

题解：https://www.acwing.com/solution/content/49316/

```c++
#include <cstdio>
#include <cstring>
#include <iostream>
#include <algorithm>

using LL = long long;
using std::cin, std::cout;

const int N = 210, INF = 0x3f3f3f3f;

int n, m, K;
int G[N][N];

int main()
{
    std::ios::sync_with_stdio(false);
    cin.tie(nullptr), cout.tie(nullptr);

    memset(G, 0x3f, sizeof G);
    for (int i = 0; i < N; i++) G[i][i] = 0;

    cin >> n >> m >> K;
    for (int i = 0; i < m; i++)
    {
        int a, b, w; cin >> a >> b >> w;
        G[a][b] = std::min(G[a][b], w);
    }

    for (int k = 1; k <= n; k++)
        for (int i = 1; i <= n; i++)
            for (int j = 1; j <= n; j++)
                G[i][j] = std::min(G[i][j], G[i][k] + G[k][j]);

    while (K--)
    {
        int a, b; cin >> a >> b;
        if (G[a][b] < INF / 2) cout << G[a][b] << '\n';
        else cout << "impossible" << '\n';
    }

    return 0;
}
```

#### 在线Floyd

题目：https://www.luogu.com.cn/problem/B3647

时间复杂度为$O(mn^2)$

```c++
#include <cstdio>
#include <cstring>
#include <iostream>
#include <algorithm>

using LL = long long;
using std::cin, std::cout;

const int N = 110;

int n, m;
int G[N][N];

void floyd(int a, int b)
{
    for (int i = 1; i <= n; i++)
    {
        G[i][b] = std::min(G[i][b], G[i][a] + G[a][b]);
        G[a][i] = std::min(G[a][i], G[a][b] + G[b][i]);
    }

    for (int i = 1; i <= n; i++)
        for (int j = 1; j <= n; j++)
            G[i][j] = std::min(G[i][j], G[i][a] + G[a][b] + G[b][j]);
}

int main()
{
    std::ios::sync_with_stdio(false);
    cin.tie(nullptr), cout.tie(nullptr);

    memset(G, 0x3f, sizeof G);
    for (int i = 0; i < N; i++) G[i][i] = 0;
    
    cin >> n >> m;
    for (int i = 0; i < m; i++)
    {
        int a, b, w; cin >> a >> b >> w;
        G[a][b] = G[b][a] = std::min(G[a][b], w);
        floyd(a, b), floyd(b, a);
    }

    for (int i = 1; i <= n; i++)
    {
        for (int j = 1; j <= n; j++)
            cout << G[i][j] << ' ';
        cout << '\n';
    }       

    return 0;
}
```

## 最小生成树

### Prim

时间复杂度$O(n^2 + m)$，$n$表示点数，$m$表示边数。

**模板题：**

[Prim算法求最小生成树](https://www.acwing.com/problem/content/860/)

```c++
#include <iostream>
#include <cstring>
using namespace std;

const int N = 510, INF = 0x3f3f3f3f;

int n, m;
int g[N][N], d[N];
bool st[N];

int prim()
{
	memset(d, 0x3f, sizeof d);
	d[1] = 0; // 最小生成树的第一个点
	int res = 0;
	for (int i = 0; i < n; i++)
	{
		int t = -1;
		for (int j = 1; j <= n; j++)
			if (!st[j] && (t == -1 || d[j] < d[t])) t = j;
		if (d[t] == INF) return INF;
		st[t] = true;
		res += d[t];
		for (int j = 1; j <= n; j++)
			if (!st[j]) d[j] = min(d[j], g[t][j]);
	}
	return res;
}

int main()
{
	scanf("%d%d", &n, &m);
	memset(g, 0x3f, sizeof g);
	while (m--)
	{
		int u, v, w; scanf("%d%d%d", &u, &v, &w);
		g[u][v] = g[v][u] = min(g[u][v], w);
	}

	int res = prim();
	if (res != INF) printf("%d\n", res);
	else puts("impossible");
	return 0;
}
```

1. 和Dijkstra算法类似，主要区别在于数组`d`的含义不同。在Prim算法中，数组`d`表示集合到其他点的距离。

**链式前向星：**

```c++
#include <cstring>
#include <iostream>

using namespace std;

const int N = 510, M = 2e5 + 10, INF = 0x3f3f3f3f;

int n, m;
int H[N];
struct ListNode
{
    int v, w, n;
} L[M]; int idx = 1;

void Add(int a, int b, int w)
{
    L[idx].v = b, L[idx].w = w, L[idx].n = H[a], H[a] = idx++;
}

bool st[N];
int D[N];
int Prim()
{
    memset(D, 0x3f, sizeof D);
    D[1] = 0;

    int res = 0;
    for (int i = 1; i <= n; i++)
    {
        int cur = -1;
        for (int j = 1; j <= n; j++)
            if (!st[j] && (cur == -1 || D[j] < D[cur]))
                cur = j;
        
        if (D[cur] == INF) return INF;

        st[cur] = true;
        res += D[cur];

        for (int p = H[cur]; p; p = L[p].n)
        {
            int v = L[p].v, w = L[p].w;
            if (!st[v] && w < D[v]) 
                D[v] = w;
        }
    }

    return res;
}

int main()
{
    scanf("%d%d", &n, &m);
    for (int i = 1; i <= m; i++)
    {
        int a, b, w; scanf("%d%d%d", &a, &b, &w);
        Add(a, b, w), Add(b, a, w);
    }

    int res = Prim();

    if (res == INF) puts("impossible");
    else printf("%d\n", res);

    return 0;
}
```



### Kruskal

时间复杂度$O(m\log_2m)$，$n$表示点数，$m$表示边数。

**模板题：**

[Kruskal算法求最小生成树](https://www.acwing.com/problem/content/861/)

```c++
#include <iostream>
#include <algorithm>

using namespace std;

const int N = 1e5 + 10, M = 2e5 + 10, INF = 0x3f3f3f3f;

int n, m;
struct Edge
{
    int a, b, w;
    bool operator< (const Edge& ct) const
    {
        return w < ct.w;
    }
} E[M];

int P[N];
int Find(int x)
{
    if (x != P[x]) P[x] = Find(P[x]);
    return P[x];
}

void Merge(int a, int b)
{
    a = Find(a), b = Find(b);
    P[a] = b;
}

int Kruskal()
{
    sort(E + 1, E + 1 + m);
    for (int i = 1; i <= n; i++) P[i] = i;

    int res = 0, cnt = 0;
    for (int i = 1; i <= m; i++)
    {
        int a = E[i].a, b = E[i].b, w = E[i].w;
        a = Find(a), b = Find(b);
        if (a != b)
        {
            res += w;
            Merge(a, b);
            if (++cnt == n - 1) return res;
        }
    }

    return INF;
}

int main()
{
    scanf("%d%d", &n, &m);
    for (int i = 1; i <= m; i++)
        scanf("%d%d%d", &E[i].a, &E[i].b, &E[i].w);

    int res = Kruskal();

    if (res == INF) puts("impossible");
    else printf("%d\n", res);
    
    return 0;
}
```

1. Kruskal算法适用于稀疏图，Prim算法适用于稠密图。

## 二分图

### 染色法

时间复杂度$O(n + m)$，$n$表示点数，$m$表示边数。

**模板题：**

[染色法判定二分图](https://www.acwing.com/problem/content/862/)

题解：https://www.acwing.com/solution/content/105874/

```c++
#include <cstdio>
#include <cstring>
#include <iostream>
#include <algorithm>

using namespace std;

const int N = 1e5 + 10, M = 2e5 + 10;

int n, m;
int H[N];
struct EdgeNode
{
    int v, n;
} E[M]; int idx;

void InitG()
{
    idx = 1;
}

void Add(int a, int b)
{
    E[idx] = {b, H[a]}, H[a] = idx++;
}

int color[N];
int Q[N], hh, tt;

bool BFS(int start)
{
    hh = 0, tt = - 1;
    Q[++tt] = start;
    color[start] = 1;

    while (hh <= tt)
    {
        int u = Q[hh++];
        int c = color[u];

        for (int p = H[u]; p; p = E[p].n)
        {
            int v = E[p].v;
            if (color[v])
            {
                if (color[v] == c) return false;
            }
            else
            {
                color[v] = -c;
                Q[++tt] = v;
            }
        }
    }

    return true;
}

int main()
{
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    InitG();

    cin >> n >> m;
    for (int i = 0; i < m; i++)
    {
        int a, b; cin >> a >> b;
        Add(a, b), Add(b, a);
    }

    bool res = true;
    for (int i = 1; i <= n; i++)
        if (!color[i])
            if (!BFS(i))
            {
                res = false;
                break;
            }

    cout << (res ? "Yes" : "No") << endl;

    return 0;
}
```

### 匈牙利算法

时间复杂度$O(nm)$，$n$表示点数，$m$表示边数。

[二分图的最大匹配](https://www.acwing.com/problem/content/863/)

```c++
#include <cstdio>
#include <cstring>
#include <iostream>
#include <algorithm>

using namespace std;

const int N = 510, M = 2e5 + 10;

int n1, n2, m;
int H[N];
struct EdgeNode
{
    int v, n;
} E[M]; int idx;

void InitG()
{
    idx = 1;
}

void Add(int a, int b)
{
    E[idx] = {b, H[a]}, H[a] = idx++;
}

bool st[N];   // 表示单次分配过程中是否被预定
int match[N]; // 由右边点集映射左边点集的已经完成的分配

bool Find(int u)
{
    for (int p = H[u]; p; p = E[p].n)
    {
        int v = E[p].v;
        if (!st[v])
        {
            st[v] = true;
            if (match[v] == 0 || Find(match[v]))
            {
                match[v] = u;
                return true;
            }
        }
    }

    return false;
}

int main()
{
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    InitG();

    cin >> n1 >> n2 >> m;
    for (int i = 0; i < m; i++)
    {
        int a, b; cin >> a >> b;
        Add(a, b);
    }

    int res = 0;
    for (int i = 1; i <= n1; i++)
    {
        memset(st, 0, sizeof st);
        res += Find(i);
    }

    cout << res << '\n';

    return 0;
}
```
