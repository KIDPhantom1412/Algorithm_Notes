### 64位整数乘法

题目：https://www.acwing.com/problem/content/description/92/

```c++
#include <cstdio>
#include <cstring>
#include <iostream>
#include <algorithm>

using std::cin, std::cout;
using LL = long long;

int main()
{
    std::ios::sync_with_stdio(false);
    cin.tie(nullptr), cout.tie(nullptr);

    LL a, b, p; cin >> a >> b >> p;
    LL res = 0;
    while (b)
    {
        if (b & 1) res = (res + a) % p;
        a = (a + a) % p;
        b >>= 1;
    }

    cout << res << '\n';

    return 0;
}
```

## 递推与递归

### 费解的开关

题目：https://www.acwing.com/problem/content/description/97/

枚举第一行的情况，之后的第$i + 1$行根据第$i$行递推。

```c++
#include <cstdio>
#include <cstring>
#include <iostream>
#include <algorithm>

using std::cin, std::cout;
using LL = long long;

const int N = 6;

char G[N][N], start[N][N];
int dx[5] = {-1, 0, 1, 0, 0}, dy[5] = {0, 1, 0, -1, 0};

void turn(int x, int y)
{
    for (int i = 0; i < 5; i++)
    {
        int nx = x + dx[i], ny = y + dy[i];
        if (nx < 0 || nx >= 5 || ny < 0 || ny >= 5) continue;
        G[nx][ny] ^= 1;
    }
}

int recur(int s)
{
    memcpy(G, start, sizeof G);
    int cnt = 0;

    for (int i = 0; i < 5; i++)
        if (s >> i & 1)
        {
            turn(0, i);
            cnt++;
        }

    for (int i = 1; i < 5; i++)
        for (int j = 0; j < 5; j++)
            if (G[i - 1][j] == '0')
            {
                turn(i, j);
                if (++cnt > 6) return -1;
            }
    
    for (int i = 0; i < 5; i++)
        if (G[4][i] == '0')
            return -1;

    return cnt;
}

void solve()
{
    for (int i = 0; i < 5; i++) cin >> start[i];

    int res = 10;
    for (int s = 0; s < 32; s++)
        if (int cnt = recur(s); cnt >= 0)
            res = std::min(res, cnt);

    cout << (res > 6 ? -1 : res) << '\n';
}

int main()
{
    std::ios::sync_with_stdio(false);
    cin.tie(nullptr), cout.tie(nullptr);

    int T; cin >> T;
    while (T--)
        solve();

    return 0;
}
```

### 约数之和

题目：https://www.acwing.com/problem/content/description/99/

```c++
#include <cstdio>
#include <cstring>
#include <iostream>
#include <algorithm>

using std::cin, std::cout;
using LL = long long;

const int MOD = 9901;

int FP(int a, int b)
{
    int res = 1;
    a %= MOD;
    while (b)
    {
        if (b & 1) res = (res * a) % MOD;
        a = (a * a) % MOD;
        b >>= 1;
    }
    return res;
}

int sum(int p, int k)
{
    if (k == 1) return 1;
    if (k % 2) return (sum(p, k - 1) + FP(p, k - 1)) % MOD;
    else return ((1 + FP(p, k / 2)) * sum(p, k / 2)) % MOD;
}

int main()
{
    std::ios::sync_with_stdio(false);
    cin.tie(nullptr), cout.tie(nullptr);

    int a, b; cin >> a >> b;
    int res = 1;
    for (int i = 2; i * i <= a; i++)
        if (a % i == 0)
        {
            int c = 0;
            while (a % i == 0)
            {
                a /= i;
                c++;
            }
            res = (res * sum(i, b * c + 1)) % MOD;
        }
    if (a > 1) res = (res * sum(a, b + 1)) % MOD;
    if (a == 0) res = 0;

    cout << res << '\n';
    return 0;
}
```

### 分形之城

题目：https://www.acwing.com/problem/content/100/

题解：https://www.acwing.com/solution/content/16524/

```c++
#include <cstdio>
#include <cstring>
#include <cmath>
#include <iostream>
#include <algorithm>

using std::cin, std::cout;
using LL = long long;

int n;
LL a, b;

struct Point
{
    int x, y;
};

Point recur(int n, LL u)
{
    if (n == 0) return {0, 0};

    LL len = 1 << (n - 1), cnt = 1LL << (n * 2 - 2);
    int group = (u - 1) / cnt;
    Point p = recur(n - 1, (u - 1) % cnt + 1);
    switch (group)
    {
    case 0: return {len - 1 - p.y, len - 1 - p.x + len};
    case 1: return {p.x + len, p.y + len};
    case 2: return {p.x + len, p.y};
    case 3: return {p.y, p.x};
    }
}

void solve()
{
    cin >> n >> a >> b;
    Point pa = recur(n, a), pb = recur(n, b);
    LL dx = pa.x - pb.x, dy = pa.y - pb.y;
    cout << LL(round(sqrt(dx * dx + dy * dy) * 10)) << '\n';
}

int main()
{
    std::ios::sync_with_stdio(false);
    cin.tie(nullptr), cout.tie(nullptr);

    int T; cin >> T;
    while (T--)
        solve();

    return 0;
}
```

## 前缀和与差分

### 激光炸弹

题目：https://www.acwing.com/problem/content/description/101/

```c++
#include <cstdio>
#include <cstring>
#include <iostream>
#include <algorithm>

using std::cin, std::cout;
using LL = long long;

const int N = 5010;

int n, r;
int ps[N][N];

int main()
{
    std::ios::sync_with_stdio(false);
    cin.tie(nullptr), cout.tie(nullptr);

    cin >> n >> r;
    int m = 0;
    for (int i = 0; i < n; i++)
    {
        int x, y, w; cin >> x >> y >> w;
        x++, y++;
        m = std::max({m, x, y});
        ps[x][y] += w;
    }

    for (int i = 1; i <= m; i++)
        for (int j = 1; j <= m; j++)
            ps[i][j] += ps[i - 1][j] + ps[i][j - 1] - ps[i - 1][j - 1];

    
    if (r > m) 
    {
        cout << ps[m][m] << '\n';
        return 0;
    }

    int res = 0;
    for (int i = r; i <= m; i++)
        for (int j = r; j <= m; j++)
            res = std::max(res, ps[i][j] - ps[i - r][j] - ps[i][j - r] + ps[i - r][j - r]);

    cout << res << '\n';
    return 0;
}
```

### 增减序列

题目：https://www.acwing.com/problem/content/description/102/

《算法竞赛进阶指南》的0x03中有讲解。原地修改可以通过倒序计算差分数组。

```c++
#include <cstdio>
#include <cstring>
#include <iostream>
#include <algorithm>

using std::cin, std::cout;
using LL = long long;

const int N = 1e5 + 10;

int n, B[N];

int main()
{
    std::ios::sync_with_stdio(false);
    cin.tie(nullptr), cout.tie(nullptr);

    cin >> n;
    for (int i = 1; i <= n; i++) cin >> B[i];
    for (int i = n; i >= 1; i--) B[i] -= B[i - 1];

    LL p = 0, q = 0;
    for (int i = 2; i <= n; i++)
    {
        if (B[i] > 0) p += B[i];
        else q -= B[i];
    }

    cout << std::max(p, q) << '\n';
    cout << std::abs(p - q) + 1 << '\n';
    return 0;
}
```

## 二分

### 最佳牛围栏

题目：https://www.acwing.com/problem/content/description/104/

题解：https://www.acwing.com/solution/content/1148/

```c++
#include <cstdio>
#include <cstring>
#include <iostream>
#include <algorithm>

using std::cin, std::cout;
using LL = long long;

const int N = 1e5 + 10;
const double INF = 1e20, eps = 1e-5;

int n, m;
double cow[N];
double ps[N];

bool check(double avg)
{
    for (int i = 1; i <= n; i++)
    {
        ps[i] = ps[i - 1] + (cow[i] - avg);
    }
    double mn = INF;
    for (int i = m; i <= n; i++)
    {
        mn = std::min(mn, ps[i - m]);
        if (ps[i] - mn > 0) return true;
    }
    return false;
}

int main()
{
    std::ios::sync_with_stdio(false);
    cin.tie(nullptr), cout.tie(nullptr);

    cin >> n >> m;
    double l = INF, r = 0;
    for (int i = 1; i <= n; i++)
    {
        cin >> cow[i];
        l = std::min(l, cow[i]);
        r = std::max(r, cow[i]);
    }

    while (r - l > eps)
    {
        double mid = (l + r) / 2;
        if (check(mid)) l = mid;
        else r = mid;
    }

    cout << int(1000 * r) << '\n';
    return 0;
}
```

### 特殊排序

题目：https://www.acwing.com/problem/content/description/115/

本题的核心在于想明白，当$res[mid] < i$的时候，我们一定可以插入到$[l, mid]$之间。

```c++
// Forward declaration of compare API.
// bool compare(int a, int b);
// return bool means whether a is less than b.

class Solution {
public:
    vector<int> specialSort(int N) {
        vector<int> res(1, 1);
        for (int i = 2; i <= N; i++)
        {
            int l = 0, r = res.size();
            while (l < r)
            {
                int mid = l + r >> 1;
                if (compare(i, res[mid])) r = mid;
                else l = mid + 1;
            }
            res.insert(res.begin() + l, i);
        }
        return res;
    }
};
```

## 排序

### 七夕祭

题目：https://www.acwing.com/problem/content/description/107/

题解：https://www.acwing.com/solution/content/52804/

```c++
#include <cstdio>
#include <cstring>
#include <iostream>
#include <algorithm>

using std::cin, std::cout;
using LL = long long;

const int N = 1e5 + 10;

int n, m, T;
int row[N], col[N], ps[N], C[N];

LL solve(int A[], int n)
{
    for (int i = 1; i <= n; i++) ps[i] = ps[i - 1] + A[i];

    if (ps[n] % n) return -1;
    int avg = ps[n] / n;

    for (int i = 1; i < n; i++) C[i] = i * avg - ps[i];
    C[n] = 0;

    std::sort(C + 1, C + 1 + n);

    int mid = C[(n + 1) / 2];
    LL res = 0;
    for (int i = 1; i <= n; i++) res += std::abs(C[i] - mid);
    return res;
}

int main()
{
    std::ios::sync_with_stdio(false);
    cin.tie(nullptr), cout.tie(nullptr);

    cin >> n >> m >> T;
    for (int i = 0; i < T; i++)
    {
        int x, y; cin >> x >> y;
        row[x]++, col[y]++;
    }

    LL a = solve(row, n);
    LL b = solve(col, m);

    if      (a != -1 && b != -1) cout << "both " << a + b << '\n';
    else if (a != -1) cout << "row " << a << '\n';
    else if (b != -1) cout << "column " << b << '\n';
    else cout << "impossible" << '\n';

    return 0;
}
```

### 动态中位数

题目：https://www.acwing.com/problem/content/description/108/

```c++
#include <cstdio>
#include <cstring>
#include <iostream>
#include <algorithm>
#include <queue>

using std::cin, std::cout;
using LL = long long;

void solve()
{
    int n, T; cin >> T >> n;
    cout << T << ' ' << (n + 1) / 2 << '\n';

    int cnt = 0;
    std::priority_queue<int> dn;
    std::priority_queue<int, std::vector<int>, std::greater<int>> up;
    for (int i = 1; i <= n; i++)
    {
        int x; cin >> x;
        if (up.empty() || x >= up.top()) up.push(x);
        else dn.push(x);
        
        if (up.size() > dn.size() + 1)
        {
            dn.push(up.top());
            up.pop();
        }
        else if (up.size() < dn.size())
        {
            up.push(dn.top());
            dn.pop();
        }

        if (i % 2) 
        {
            cout << up.top() << ' ';
            if (++cnt % 10 == 0) cout << '\n';
        }
    }
    if (cnt % 10) cout << '\n';
}

int main()
{
    std::ios::sync_with_stdio(false);
    cin.tie(nullptr), cout.tie(nullptr);

    int T; cin >> T;
    while (T--)
        solve();

    return 0;
}
```

### 超快速排序

题目：https://www.acwing.com/problem/content/description/109/

只要逆序对不为0，就必然可以找到一次相邻交换使逆序对减少1，而且最多只能减少1。所以可以找到一次相邻交换和减少1个逆序对的映射方式。

```c++
#include <cstdio>
#include <cstring>
#include <iostream>
#include <algorithm>

using std::cin, std::cout;
using LL = long long;

const int N = 5e5 + 10;

int n;
LL A[N], B[N];

LL mergeSort(int l, int r)
{
    if (l == r) return 0;

    LL res = 0;
    int mid = l + r >> 1;
    res += mergeSort(l, mid);
    res += mergeSort(mid + 1, r);

    int i = l, j = mid + 1, k = l;
    while (i <= mid && j <= r)
    {
        if (A[i] <= A[j]) B[k++] = A[i++];
        else
        {
            res += mid - i + 1;
            B[k++] = A[j++];
        }
    }
    while (i <= mid) B[k++] = A[i++];
    while (j <= r) B[k++] = A[j++];
    for (int k = l; k <= r; k++) A[k] = B[k];
    return res;
}

int main()
{
    std::ios::sync_with_stdio(false);
    cin.tie(nullptr), cout.tie(nullptr);

    while (cin >> n, n)
    {
        for (int i = 1; i <= n; i++) cin >> A[i];
        cout << mergeSort(1, n) << '\n';
    }

    return 0;
}
```

## RMQ

### 天才的记忆

题目：https://www.acwing.com/problem/content/description/1275/

```c++
#include <cstdio>
#include <cstring>
#include <cmath>
#include <iostream>
#include <algorithm>

using std::cin, std::cout;
using LL = long long;

const int N = 2e5 + 10, M = 20;

int n, m;
int f[N][M];

int main()
{
    std::ios::sync_with_stdio(false);
    cin.tie(nullptr), cout.tie(nullptr);

    cin >> n;
    for (int i = 1; i <= n; i++) cin >> f[i][0];

    int s = log2(n);
    for (int j = 1; j <= s; j++)
        for (int i = 1; i + (1 << j - 1) <= n; i++)
            f[i][j] = std::max(f[i][j - 1], f[i + (1 << j - 1)][j - 1]);
    
    cin >> m;
    while (m--)
    {
        int l, r; cin >> l >> r;
        int k = log2(r - l + 1);
        int res = std::max(f[l][k], f[r - (1 << k) + 1][k]);
        cout << res << '\n';
    }

    return 0;
}
```
