## 基环树DP

### 岛屿

题目：https://www.acwing.com/problem/content/description/360/

题解：https://www.acwing.com/solution/content/166480/

```c++
#include <cstdio>
#include <iostream>
#include <algorithm>
#include <numeric>
#include <vector>

using LL = long long;
using std::cin, std::cout;
using std::vector;

const int N = int(1e6) + 5, M = N * 2;

int n;
int H[N];
struct
{
    int v, w, n;
} E[M];
int idx = 1;

void add(int a, int b, int w)
{
    E[++idx] = { b, w, H[a] }, H[a] = idx;
}

bool st[N];
vector<vector<int>> circles;
int fw[N];
int stk[N], top;
bool inStk[N];
LL sv[N];

void findCircle(int u, int from)
{
    inStk[u] = st[u] = true;
    stk[++top] = u;

    for (int i = H[u]; i; i = E[i].n)
        if (i != (from ^ 1))
        {
            int v = E[i].v, w = E[i].w;
            fw[v] = w;
            if (!st[v])
            {
                findCircle(v, i);
            }
            else if (inStk[v])
            {
                vector<int> circle;
                LL sum = w;
                for (int j = top; stk[j] != v; j--)
                {
                    circle.push_back(stk[j]);
                    sv[stk[j]] = sum;
                    sum += fw[stk[j]];
                }
                circle.push_back(v);
                sv[v] = sum;
                circles.push_back(circle);
            }
        }

    inStk[u] = false;
    top--;
}

LL getHeight(int u, LL& mx)
{
    st[u] = true;

    LL h1 = 0, h2 = 0;
    for (int i = H[u]; i; i = E[i].n)
    {
        int v = E[i].v, w = E[i].w;
        if (!st[v])
        {
            LL h = getHeight(v, mx) + w;
            if (h > h1) h2 = h1, h1 = h;
            else if (h > h2) h2 = h;
        }
    }

    mx = std::max(mx, h1 + h2);
    return h1;
}

int main()
{
    std::ios::sync_with_stdio(false);
    cin.tie(nullptr), cout.tie(nullptr);

    cin >> n;
    for (int i = 1; i <= n; i++)
    {
        int a, l; cin >> a >> l;
        add(i, a, l), add(a, i, l);
    }

    for (int i = 1; i <= n; i++)
        if (!st[i])
            findCircle(i, -1);

    std::fill_n(st, N, false);
    for (auto& circle : circles)
        for (int u : circle)
        {
            st[u] = true;
        }

    LL res = 0;
    for (auto& circle : circles)
    {
        LL mx = 0;
        static LL ps[N * 2], h[N * 2];
        int cnt = circle.size();
        for (int i = 0; i < cnt; i++)
        {
            int u = circle[i];
            h[i] = getHeight(u, mx);
            ps[i] = sv[u];
        }
        std::copy(h, h + cnt, h + cnt);
        for (int i = 0; i < cnt; i++) ps[i + cnt] = ps[cnt - 1] + ps[i];

        static int Q[N * 2];
        int hh = 0, tt = 0;
        for (int i = 0; i < cnt * 2; i++)
        {
            if (hh != tt && i - Q[hh] + 1 > cnt) hh++;
            if (hh != tt) mx = std::max(mx, ps[i] + h[i] - ps[Q[hh]] + h[Q[hh]]);
            while (hh != tt && h[Q[tt - 1]] - ps[Q[tt - 1]] <= h[i] - ps[i]) tt--;
            Q[tt++] = i;
        }

        res += mx;
    }
    
    cout << res << '\n';

    return 0;
}
```

### 骑士

题目：https://www.acwing.com/problem/content/description/1082/

题解：https://www.acwing.com/solution/content/166497/

```c++
#include <cstdio>
#include <iostream>
#include <algorithm>
#include <numeric>

using LL = long long;
using std::cin, std::cout;

const int N = int(1e6) + 5;
const LL INF = LL(1e13);  // 1e6 * 1e6是无穷值

int n;
int H[N], ce[N];
struct
{
    int v, n;
} E[N];
int idx;

void add(int a, int b)
{
    E[++idx] = { b, H[a] }, H[a] = idx;
}

LL f1[N][2], f2[N][2];

void dp(int u, int hater, int de, LL f[][2])
{
    for (int i = H[u]; i; i = E[i].n)
        if (int v = E[i].v; i != de)
        {
            dp(v, hater, de, f);
            f[u][0] += std::max(f[v][0], f[v][1]);
        }

    f[u][1] = -INF;
    if (u != hater)
    {
        f[u][1] = ce[u];
        for (int i = H[u]; i; i = E[i].n)
            if (int v = E[i].v; i != de)
            {
                f[u][1] += f[v][0];
            }
    }
}

bool inStk[N], st[N];
LL res;

void dfs(int u)
{
    inStk[u] = st[u] = true;

    for (int i = H[u]; i; i = E[i].n)
    {
        int v = E[i].v;
        if (!st[v])
        {
            dfs(v);
        }
        else if (inStk[v])  // 破开u->v这条边
        {
            dp(v, 0, i, f1);
            dp(v, u, i, f2);
            res += std::max(f1[v][0], f2[v][1]);
        }
    }

    inStk[u] = false;
}

int main()
{
    std::ios::sync_with_stdio(false);
    cin.tie(nullptr), cout.tie(nullptr);

    cin >> n;
    for (int i = 1; i <= n; i++)
    {
        int a, b; cin >> a >> b;
        add(b, i);  // 必须反向建树，建成外向树
        ce[i] = a;
    }

    for (int i = 1; i <= n; i++)
        if (!st[i])
            dfs(i);

    cout << res << '\n';

    return 0;
}
```

### 创世纪

题目：https://www.acwing.com/problem/content/description/361/

题解：https://www.acwing.com/solution/content/166527/

```c++
#include <cstdio>
#include <iostream>
#include <algorithm>
#include <numeric>

using LL = long long;
using std::cin, std::cout;

const int N = int(1e6) + 5;

int n;
int H[N];
struct
{
    int v, n;
} E[N];
int idx;

void add(int a, int b)
{
    E[++idx] = { b, H[a] }, H[a] = idx;
}

void dp(int u, int de, int mc, int f[][2])
{
    for (int i = H[u]; i; i = E[i].n)
        if (int v = E[i].v; i != de)
        {
            dp(v, de, mc, f);
            f[u][0] += std::max(f[v][0], f[v][1]);
        }

    if (u == mc)
    {
        f[u][1] = f[u][0] + 1;
        f[u][0] = -N;
    }
    else
    {
        for (int i = H[u]; i; i = E[i].n)
            if (int v = E[i].v; i != de)
            {
                f[u][1] = std::max(f[u][1], f[u][0] - std::max(f[v][0], f[v][1]) + f[v][0] + 1);
            }
    }
}

bool st[N], inStk[N];
int f1[N][2], f2[N][2];
int res;

void dfs(int u)
{
    st[u] = inStk[u] = true;

    for (int i = H[u]; i; i = E[i].n)
    {
        int v = E[i].v;
        if (!st[v]) dfs(v);
        else if (inStk[v])
        {
            dp(v, i, 0, f1);
            dp(v, i, u, f2);
            res += std::max({f1[v][0], f1[v][1], f2[v][0]});
        }
    }

    inStk[u] = false;
}

int main()
{
    std::ios::sync_with_stdio(false);
    cin.tie(nullptr), cout.tie(nullptr);

    cin >> n;
    for (int i = 1; i <= n; i++)
    {
        int a; cin >> a;
        add(a, i);
    }

    for (int i = 1; i <= n; i++)
        if (!st[i])
            dfs(i);

    cout << res << '\n';

    return 0;
}
```

### 诗人小G

题目：https://www.acwing.com/problem/content/description/306/

题解：https://www.acwing.com/solution/content/166943/

```c++
#include <cstdio>
#include <iostream>
#include <algorithm>
#include <numeric>

using LL = long long;
using LD = long double;
using std::cin, std::cout;

const int N = int(1e5) + 5;

int n, L, P;
std::string S[N];
LL ps[N];
LD f[N];
int opt[N];
struct
{
    int j, l, r;
} Q[N];
int hh, tt;

LD get(int j, int i)
{
    LD res = 1, t = std::abs(ps[i] - ps[j] + i - j - 1 - L);
    for (int k = P; k--; ) res *= t;
    return res += f[j];
}

void update(int i)
{
    int mn = N;
    while (hh != tt &&
        get(i, Q[tt - 1].l) <= get(Q[tt - 1].j, Q[tt - 1].l))
    {
        mn = Q[--tt].l;
    }
    if (hh != tt && 
        get(i, Q[tt - 1].r) <= get(Q[tt - 1].j, Q[tt - 1].r))
    {
        int l = Q[tt - 1].l, r = Q[tt - 1].r;
        while (l < r)
        {
            int mid = (l + r) / 2;
            if (get(i, mid) <= get(Q[tt - 1].j, mid)) r = mid;
            else l = mid + 1;
        }
        Q[tt - 1].r = l - 1;
        mn = l;
    }

    if (mn != N) Q[tt++] = { i, mn, n };
}

void solve()
{
    cin >> n >> L >> P;
    for (int i = n; i >= 1; i--) cin >> S[i];
    for (int i = 1; i <= n; i++) ps[i] = ps[i - 1] + S[i].size();

    hh = tt = 0;
    Q[tt++] = { 0, 1, n };
    for (int i = 1; i <= n; i++)
    {
        opt[i] = Q[hh].j;
        f[i] = get(opt[i], i);
        if (i == Q[hh].r) hh++;
        Q[hh].l = i + 1;
        update(i);
    }

    if (f[n] > 1e18)
    {
        cout << "Too hard to arrange" << '\n';
        return;
    }

    cout << LL(f[n]) << '\n';
    for (int i = n; i; i = opt[i])
    {
        for (int j = i; j > opt[i]; j--)
        {
            cout << S[j];
            if (j != opt[i] + 1) cout << ' ';
        }
        cout << '\n';
    }
}

int main()
{
    std::ios::sync_with_stdio(false);
    cin.tie(nullptr), cout.tie(nullptr);
    
    int t; cin >> t;
    while (t--)
    {
        solve();
        cout << "--------------------" << '\n';
    }

    return 0;
}
```

### 再探石子合并

题目：https://www.acwing.com/problem/content/2892/

题解：https://www.acwing.com/solution/content/167339/

```c++
#include <cstdio>
#include <iostream>
#include <algorithm>
#include <numeric>

using LL = long long;
using std::cin, std::cout;

// 这个INF究竟为多少，会大于INT_MAX吗？
const int N = 5010, INF = std::numeric_limits<int>::max();

int n;
int f[N][N], P[N][N], ps[N];

int main()
{
    std::ios::sync_with_stdio(false);
    cin.tie(nullptr), cout.tie(nullptr);

    cin >> n;
    for (int i = 1; i <= n; i++)
    {
        cin >> ps[i];
        ps[i] += ps[i - 1];
    }

    for (int i = 1; i <= n; i++) P[i][i] = i;
    for (int len = 2; len <= n; len++)
        for (int i = 1; i + len - 1 <= n; i++)
        {
            int j = i + len - 1;
            f[i][j] = INF;
            for (int k = P[i][j - 1]; k <= P[i + 1][j]; k++)
            {
                int t = f[i][k] + f[k + 1][j] + (ps[j] - ps[i - 1]);
                if (t <= f[i][j])
                {
                    f[i][j] = t;
                    P[i][j] = k;
                }
            }
        }

    cout << f[1][n] << '\n';

    return 0;
}
```

## 插头DP

### 插头DP

题目：https://www.acwing.com/problem/content/description/2937/

题解：https://www.acwing.com/solution/content/167520/

图解：https://www.luogu.com.cn/article/rx5k5drf

```c++
#include <cstdio>
#include <iostream>
#include <algorithm>
#include <numeric>

using LL = long long;
using std::cin, std::cout;

const int N = int(5e4) + 5, M = 100343;

int n, m;
char G[15][15];
int ex, ey;
int f[2][N], cnt[2];  // f存储哈希表的下标
int K[2][M];
LL V[2][M];

int find(int cur, int x)
{
    int t = x % M;
    while (K[cur][t] != -1 && K[cur][t] != x)
        if (++t == M)
            t = 0;
    return t;
}

void update(int cur, int s, LL d)
{
    int t = find(cur, s);
    if (K[cur][t] == -1)
    {
        f[cur][cnt[cur]++] = t;
        K[cur][t] = s;
        V[cur][t] = d;
    }
    else
    {
        V[cur][t] += d;
    }
}

void init(int cur)
{
    std::fill_n(K[cur], M, -1);
    cnt[cur] = 0;
}

int get(int s, int k)
{
    return s >> (k * 2) & 3;
}

int set(int x, int k)
{
    return x << (k * 2);
}

int main()
{
    std::ios::sync_with_stdio(false);
    cin.tie(nullptr), cout.tie(nullptr);

    cin >> n >> m;
    for (int i = 0; i < n; i++) cin >> G[i];
    for (int i = 0; i < n; i++)
        for (int j = 0; j < m; j++)
        {
            if (G[i][j] == '*') G[i][j] = 0;
            if (G[i][j]) ex = i, ey = j;
        }

    LL res = 0;
    int cur = 0;
    init(cur);
    update(cur, 0, 1);
    for (int i = 0; i < n; i++)
    {
        for (int k = 0; k < cnt[cur]; k++)
        {
            K[cur][f[cur][k]] <<= 2;
        }

        for (int j = 0; j < m; j++)
        {
            int pre = cur;
            cur = cur ^ 1;
            init(cur);
            for (int k = 0; k < cnt[pre]; k++)
            {
                int s = K[pre][f[pre][k]];
                LL v = V[pre][f[pre][k]];
                int x = get(s, j), y = get(s, j + 1);
                if (!G[i][j])
                {
                    if (!x && !y) update(cur, s, v);
                }
                else if (!x && !y)
                {
                    if (G[i + 1][j] && G[i][j + 1]) 
                        update(cur, s + set(1, j) + set(2, j + 1), v);
                }
                else if (!x && y)
                {
                    if (G[i + 1][j]) update(cur, s - set(y, j + 1) + set(y, j), v);
                    if (G[i][j + 1]) update(cur, s, v);
                }
                else if (x && !y)
                {
                    if (G[i + 1][j]) update(cur, s, v);
                    if (G[i][j + 1]) update(cur, s - set(x, j) + set(x, j + 1), v);
                }
                else if (x == 1 && y == 1)
                {
                    for (int r = j + 2, b = 1; ; r++)
                    {
                        int z = get(s, r);
                        if (z == 1) b++;
                        else if (z == 2)
                        {
                            if (--b == 0)
                            {
                                update(cur, s - set(1, j) - set(1, j + 1) - set(1, r), v);
                                break;
                            }
                        }
                    }
                }
                else if (x == 1 && y == 2)
                {
                    if (i == ex && j == ey) res += v;
                }
                else if (x == 2 && y == 1)
                {
                    update(cur, s - set(2, j) - set(1, j + 1), v);
                }
                else // x == 2 && y == 2
                {
                    for (int r = j - 1, b = 1; ; r--)
                    {
                        int z = get(s, r);
                        if (z == 2) b++;
                        else if (z == 1)
                        {
                            if (--b == 0)
                            {
                                update(cur, s - set(2, j) - set(2, j + 1) + set(1, r), v);
                                break;
                            }
                        }
                    }
                }
            }
        }
    }

    cout << res << '\n';

    return 0;
}
```

### 神奇游乐园

题目：https://www.acwing.com/problem/content/2146/

题解：https://www.acwing.com/solution/content/167712/

本题必须要考虑闭合的时候只有一个回路闭合。

```c++
#include <cstdio>
#include <iostream>
#include <algorithm>
#include <numeric>

using LL = long long;
using std::cin, std::cout;

const int N = 150, M = 307, INF = int(1e6);

int n, m;
int G[110][10];
int f[2][N], cnt[N];
int K[2][M], V[2][M];

void init(int cur)
{
    cnt[cur] = 0;
    std::fill_n(K[cur], M, -1);
}

int find(int cur, int s)
{
    int t = s % M;
    while (K[cur][t] != -1 && K[cur][t] != s)
        if (++t == M)
            t = 0;
    return t;
}

void update(int cur, int s, int v)
{
    int t = find(cur, s);
    if (K[cur][t] == -1)
    {
        K[cur][t] = s;
        V[cur][t] = v;
        f[cur][cnt[cur]++] = t;
    }
    else
    {
        V[cur][t] = std::max(V[cur][t], v);
    }
}

int get(int s, int k)
{
    return s >> k * 2 & 3;
}

int set(int x, int k)
{
    return x << k * 2;
}

int main()
{
    std::ios::sync_with_stdio(false);
    cin.tie(nullptr), cout.tie(nullptr);

    cin >> n >> m;
    for (int i = 0; i < n; i++)
        for (int j = 0; j < m; j++)
            cin >> G[i][j];

    int res = -INF;
    int cur = 0;
    init(cur);
    update(cur, 0, 0);
    for (int i = 0; i < n; i++)
    {
        for (int k = 0; k < cnt[cur]; k++)
            K[cur][f[cur][k]] <<= 2;

        for (int j = 0; j < m; j++)
        {
            int pre = cur;
            cur ^= 1;
            init(cur);

            for (int k = 0; k < cnt[pre]; k++)
            {
                int s = K[pre][f[pre][k]], v = V[pre][f[pre][k]];
                int d = G[i][j];
                int x = get(s, j), y = get(s, j + 1);
                if (!x && !y)
                {
                    update(cur, s, v);
                    if (i < n - 1 && j < m - 1)
                    {
                        update(cur, s + set(1, j) + set(2, j + 1), v + d);
                    }
                }
                else if (!x && y)
                {
                    if (i < n - 1) update(cur, s - set(y, j + 1) + set(y, j), v + d);
                    if (j < m - 1) update(cur, s, v + d);
                }
                else if (x && !y)
                {
                    if (i < n - 1) update(cur, s, v + d);
                    if (j < m - 1) update(cur, s - set(x, j) + set(x, j + 1), v + d);
                }
                else if (x == 1 && y == 1)
                {
                    for (int r = j + 2, b = 1; ; r++)
                    {
                        int z = get(s, r);
                        if (z == 1) b++;
                        else if (z == 2)
                        {
                            if (--b == 0)
                            {
                                update(cur, s - set(x, j) - set(y, j + 1) - set(1, r), v + d);
                                break;
                            }
                        }
                    }
                }
                else if (x == 1 && y == 2)
                {
                    if (s == set(x, j) + set(y, j + 1))
                    {
                        res = std::max(res, v + d);
                    }
                }
                else if (x == 2 && y == 1)
                {
                    update(cur, s - set(x, j) - set(y, j + 1), v + d);
                }
                else // x == 2 && y == 2
                {
                    for (int r = j - 1, b = 1; ; r--)
                    {
                        int z = get(s, r);
                        if (z == 2) b++;
                        else if (z == 1)
                        {
                            if (--b == 0)
                            {
                                update(cur, s - set(x, j) - set(y, j + 1) + set(1, r), v + d);
                                break;
                            }
                        }
                    }
                }
            }
        }
    }

    cout << res << '\n';

    return 0;
}
```

### 地板

题目：https://www.acwing.com/problem/content/description/2646/

题解：https://www.acwing.com/solution/content/167819/

```c++
#include <cstdio>
#include <iostream>
#include <algorithm>
#include <numeric>

using LL = long long;
using std::cin, std::cout;

const int N = int(18e4), M = N * 4 + 7, MOD = 20110520;

int n, m;
char G[110][110];
int ex, ey;
int K[2][M], V[2][M];
int f[2][N], cnt[2];

void init(int cur)
{
    cnt[cur] = 0;
    std::fill_n(K[cur], M, -1);
}

int find(int cur, int s)
{

void update(int cur, int s, int d)

{

    cin.tie(nullptr), cout.tie(nullptr);

    cin >> n >> m;
    for (int i = 1; i <= n; i++)
    {
        std::string str; cin >> str;
        for (int j = 1; j <= str.size(); j++)
        {
            if (str[j - 1] == '_')
            {
                G[i][j] = 1;
                ex = i, ey = j;
            }
        }
    }

    if (n < m)
    {
        std::swap(n, m);
        std::swap(ex, ey);
        for (int i = 1; i <= n; i++)
            for (int j = 1; j < i; j++)
                std::swap(G[i][j], G[j][i]);
    }

    int res = 0;
    int cur = 0;
    init(cur);
    update(cur, 0, 1);
    for (int i = 1; i <= n; i++)
    {
        for (int k = 0; k < cnt[cur]; k++)
            K[cur][f[cur][k]] <<= 2;

        for (int j = 1; j <= m; j++)
        {
            int pre = cur;
            cur ^= 1;
            init(cur);

            for (int k = 0; k < cnt[pre]; k++)
            {
                int s = K[pre][f[pre][k]], v = V[pre][f[pre][k]];
                int x = get(s, j - 1), y = get(s, j);

                if (!G[i][j])
                {
                    if (!x && !y) update(cur, s, v);
                }
                else if (!x && !y)
                {
                    if (G[i + 1][j]) update(cur, s + set(1, j - 1), v);
                    if (G[i][j + 1]) update(cur, s + set(1, j), v);
                    if (G[i + 1][j] && G[i][j + 1])
                    {
                        update(cur, s + set(2, j - 1) + set(2, j), v);
                    }
                }
                else if (!x && y == 1)
                {
                    if (G[i + 1][j])
                    {
                        update(cur, s - set(y, j) + set(y, j - 1), v);
                    }
                    if (G[i][j + 1])
                    {
                        update(cur, s + set(1, j), v);
                    }
                }
                else if (!x && y == 2)
                {
                    if (i == ex && j == ey)
                    {
                        if (s == set(y, j)) res = (res + v) % MOD;
                    }
                    else
                    {
                        update(cur, s - set(y, j), v);
                        if (G[i + 1][j]) update(cur, s - set(y, j) + set(y, j - 1), v);
                    }
                }
                else if (x == 1 && !y)
                {
                    if (G[i + 1][j]) update(cur, s + set(1, j - 1), v);
                    if (G[i][j + 1]) update(cur, s - set(x, j - 1) + set(x, j), v);
                }
                else if (x == 1 && y == 1)
                {
                    if (i == ex && j == ey)
                    {
                        if (s == set(x, j - 1) + set(y, j)) res = (res + v) % MOD;
                    }
                    else
                    {
                        update(cur, s - set(x, j - 1) - set(y, j), v);
                    }
                }
                else if (x == 2 && !y)
                {
                    if (i == ex && j == ey)
                    {
                        if (s == set(x, j - 1)) res = (res + v) % MOD;
                    }
                    else
                    {
                        update(cur, s - set(x, j - 1), v);
                        if (G[i][j + 1]) update(cur, s - set(x, j - 1) + set(x, j), v);
                    }
                }
            }
        }
    }

    cout << res << '\n';

    return 0;
}
```
