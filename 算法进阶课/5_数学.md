# 数学

## 莫比乌斯反演

### promblem b

题目：https://www.acwing.com/problem/content/2704/

题解：https://www.acwing.com/solution/content/64817/

```c++
#include <cstdio>
#include <iostream>
#include <algorithm>
#include <numeric>

using LL = long long;
using std::cin, std::cout;

const int N = int(5e4) + 5;

int a, b, c, d, k;
bool st[N];
int prime[N], cnt;
int mob[N], ps[N];

void mobiusSieve()
{
    mob[1] = 1;
    for (int i = 2; i < N; i++)
    {
        if (!st[i])
        {
            prime[cnt++] = i;
            mob[i] = -1;
        }
        for (int j = 0; LL(prime[j]) * i < N; j++)
        {
            st[prime[j] * i] = true;
            if (i % prime[j] == 0) break;
            mob[prime[j] * i] = -mob[i];
        }
    }
}

int endPoint(int i, int n)
{
    return n / (n / i);
}

LL f(int a, int b)  // (0,0)->(a,b)矩形内的所有满足要求的点
{
    a /= k, b /= k;
    int n = std::min(a, b);
    LL res = 0;
    for (int l = 1, r = 1; l <= n; l = r + 1)
    {
        r = std::min({ n, endPoint(l, a), endPoint(l, b) });
        res += LL(ps[r] - ps[l - 1]) * (a / l) * (b / l);
    }
    return res;
}

void solve()
{
    cin >> a >> b >> c >> d >> k;
    cout << f(b, d) - f(b, c - 1) - f(a - 1, d) + f(a - 1, c - 1) << '\n';
}

int main()
{
    std::ios::sync_with_stdio(false);
    cin.tie(nullptr), cout.tie(nullptr);

    mobiusSieve();
    for (int i = 1; i < N; i++) ps[i] = ps[i - 1] + mob[i];

    int t; cin >> t;
    while (t--)
        solve();

    return 0;
}
```
