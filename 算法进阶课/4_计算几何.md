### 玩具

题目：https://www.acwing.com/problem/content/description/2986/

题解：https://www.acwing.com/solution/content/168370/

```c++
#include <cstdio>
#include <iostream>
#include <algorithm>
#include <numeric>

using LL = long long;
using std::cin, std::cout;

const int N = 5010;

int n, m;

struct Point
{
    int x, y;
} A[N], B[N];

LL cross(Point a, Point b)
{
    return LL(a.x) * b.y - LL(b.x) * a.y;
}

LL area(Point a, Point b, Point c)
{
    return cross({ b.x - a.x, b.y - a.y }, { c.x - a.x, c.y - a.y });
}

int find(Point p)
{
    int l = 0, r = n;
    while (l < r)
    {
        int mid = (l + r) / 2;
        if (area(B[mid], A[mid], p) > 0) r = mid;
        else l = mid + 1;
    }
    return l;
}

void solve(int t)
{
    cin >> m;
    int x1, y1, x2, y2; cin >> x1 >> y1 >> x2 >> y2;
    for (int i = 0; i < n; i++)
    {
        int u, l; cin >> u >> l;
        A[i] = { u, y1 }, B[i] = { l, y2 };
    }
    A[n] = { x2, y1 }, B[n] = { x2, y2 };

    static int cnt[N];
    std::fill_n(cnt, N, 0);
    for (int i = 0; i < m; i++)
    {
        int x, y; cin >> x >> y;
        cnt[find({ x, y })]++;
    }
    if (t) cout << '\n';
    for (int i = 0; i <= n; i++) cout << i << ": " << cnt[i] << '\n';
}

int main()
{
    std::ios::sync_with_stdio(false);
    cin.tie(nullptr), cout.tie(nullptr);

    int t = 0;
    while (cin >> n, n)
        solve(t++);

    return 0;
}
```

### 线段

题目：https://www.acwing.com/problem/content/description/2987/

题解：https://www.acwing.com/solution/content/168396/

```c++
#include <cstdio>
#include <iostream>
#include <algorithm>
#include <numeric>

using LL = long long;
using std::cin, std::cout;

const int N = 110;
const double eps = 1e-8;

int n;
struct Point
{
    double x, y;
} A[N], B[N], Q[N * 2];

int cmp(double a, double b)
{
    if (std::abs(a - b) < eps) return 0;
    return a < b ? -1 : 1;
}

int sign(double a)
{
    return cmp(a, 0);
}

bool isEqual(Point a, Point b)
{
    return !cmp(a.x, b.x) && !cmp(a.y, b.y);
}

double cross(Point a, Point b)
{
    return a.x * b.y - a.y * b.x;
}

double area(Point a, Point b, Point c)
{
    return cross({ b.x - a.x, b.y - a.y }, { c.x - a.x, c.y - a.y });
}

bool check()
{
    for (int i = 0; i < n * 2; i++)
        for (int j = 0; j < i; j++)
            if (!isEqual(Q[i], Q[j]))
            {
                bool flag = true;
                for (int k = 0; k < n; k++)
                {
                    if (sign(area(Q[i], Q[j], A[k])) * 
                        sign(area(Q[i], Q[j], B[k])) > 0)
                    {
                        flag = false;
                        break;
                    }
                }
                if (flag) return true;
            }
    return false;
}

void solve()
{
    cin >> n;
    for (int i = 0; i < n; i++)
    {
        double x1, y1, x2, y2; cin >> x1 >> y1 >> x2 >> y2;
        A[i] = { x1, y1 }, B[i] = { x2, y2 };
        Q[i * 2] = { x1, y1 }, Q[i * 2 + 1] = { x2, y2 };
    }

    cout << (check() ? "Yes!" : "No!") << '\n';
}

int main()
{
    std::ios::sync_with_stdio(false);
    cin.tie(nullptr), cout.tie(nullptr);

    int t; cin >> t;
    while (t--)
        solve();

    return 0;
}
```

### 围住奶牛

题目：https://www.acwing.com/problem/content/description/1403/

题解：https://www.acwing.com/solution/content/168552/

```c++
#include <cstdio>
#include <cmath>
#include <iostream>
#include <algorithm>
#include <numeric>

using LL = long long;
using std::cin, std::cout;

const int N = int(1e4) + 5;

int n;

struct Point
{
    double x, y;

    bool operator< (const Point& o)
    {
        return x != o.x ? x < o.x : y < o.y;
    }

    bool operator== (const Point& o)
    {
        return x == o.x && y == o.y;
    }

    Point operator- (const Point& o)
    {
        return { x - o.x, y - o.y };
    }
} P[N];

double cross(Point a, Point b)
{
    return a.x * b.y - a.y * b.x;
}

double area(Point a, Point b, Point c)
{
    return cross(b - a, c - a);
}

double getDist(Point a, Point b)
{
    double dx = a.x - b.x, dy = a.y - b.y;
    return sqrt(dx * dx + dy * dy);
}

double Andrew()
{
    std::sort(P, P + n);
    n = std::unique(P, P + n) - P;

    static bool st[N];
    static int stk[N];
    int top = 0;
    for (int i = 0; i < n; i++)
    {
        while (top >= 2 && 
            area(P[stk[top - 1]], P[stk[top]], P[i]) > 0)
        {
            st[stk[top--]] = false;
        }
        stk[++top] = i;
        st[i] = true;
    }

    st[0] = false;
    for (int i = n - 1; i >= 0; i--)
        if (!st[i])
        {
            while (top >= 2 && 
                area(P[stk[top - 1]], P[stk[top]], P[i]) > 0)
            {
                top--;
            }
            stk[++top] = i;
        }

    double res = 0;
    for (int i = 1; i < top; i++)
    {
        res += getDist(P[stk[i]], P[stk[i + 1]]);
    }
    return res;
}

int main()
{
    std::ios::sync_with_stdio(false);
    cin.tie(nullptr), cout.tie(nullptr);

    cin >> n;
    for (int i = 0; i < n; i++)
    {
        double x, y; cin >> x >> y;
        P[i] = { x, y };
    }

    printf("%.2lf\n", Andrew());
    return 0;
}
```

### 信用卡凸包

题目：https://www.acwing.com/problem/content/description/2938/

题解：https://www.acwing.com/solution/content/168584/

```c++
#include <cstdio>
#include <cmath>
#include <iostream>
#include <algorithm>
#include <numeric>

using LL = long long;
using std::cin, std::cout;

const int N = int(4e4) + 5;

Point rotate(Point v, double z)

            {
                top--;

        }

        res += getDist(P[stk[i]], P[stk[i + 1]]);

    int n; cin >> n;
    double a, b, r; cin >> a >> b >> r;
    a = a / 2 - r, b = b / 2 - r;

    static int dx[4] = { 1, -1, -1, 1 };
    static int dy[4] = { 1, 1, -1, -1 };
    for (int i = 0; i < n; i++)

            Point v = rotate({ b * dx[j], a * dy[j]}, z);
            P[cnt++] = { x + v.x, y + v.y };
        }
    }

    double res = 2 * PI * r + Andrew();
    printf("%.2lf\n", res);

    return 0;
```

## 半平面交

### 凸多边形

题目：https://www.acwing.com/problem/content/description/2805/

题解：《算法竞赛》

```c++
#include <cstdio>
#include <cmath>
#include <iostream>
#include <algorithm>
#include <numeric>
#include <vector>

using LL = long long;
using std::cin, std::cout;

const int N = 510;
const double eps = 1e-8;

struct Point
{
    double x, y;

    Point operator- (const Point& o) const
    {
        return { x - o.x, y - o.y };
    }

    Point operator+ (const Point& o) const
    {
        return { x + o.x, y + o.y };
    }

    Point operator* (double k) const
    {
        return { x * k, y * k };
    }
};

struct Line
{
    Point p, v;
    double ang;

    Line() {}
    Line(Point p, Point v) : p(p), v(v), ang(atan2(v.y, v.x)) {}

    bool operator< (const Line& o) const
    {
        return ang < o.ang;
    }
} L[N];
int cnt;

double cross(Point a, Point b)
{
    return a.x * b.y - a.y * b.x;
}

double area(Point a, Point b, Point c)
{
    return cross(b - a, c - a);
}

int sign(double v)
{
    if (std::abs(v) < eps) return 0;
    return v < 0 ? -1 : 1;
}

bool onLeft(Line l, Point p)
{
    return sign(cross(l.v, p - l.p)) > 0;
}

Point getIntersection(Line a, Line b)
{
    Point u = a.p - b.p;
    double t = cross(b.v, u) / cross(a.v, b.v);
    return a.p + a.v * t;
}

int ql[N];
Point qp[N];
int hh, tt;

void HPI()
{
    std::sort(L, L + cnt);

    ql[tt++] = 0;
    for (int i = 1; i < cnt; i++)
    {
        while (tt - hh >= 2 && !onLeft(L[i], qp[tt - 1])) tt--;
        while (tt - hh >= 2 && !onLeft(L[i], qp[hh + 1])) hh++;
        ql[tt++] = i;
        if (!sign(cross(L[i].v, L[ql[tt - 2]].v)))  // 去掉平行边
        {
            tt--;
            if (onLeft(L[ql[tt - 1]], L[i].p)) ql[tt - 1] = i;
        }
        if (tt - hh >= 2)
        {
            qp[tt - 1] = getIntersection(L[ql[tt - 1]], L[ql[tt - 2]]);
        }
    }
    while (tt - hh >= 2 && !onLeft(L[ql[hh]], qp[tt - 1])) tt--;
    qp[tt] = getIntersection(L[ql[tt - 1]], L[ql[hh]]);
}

int main()
{
    std::ios::sync_with_stdio(false);
    cin.tie(nullptr), cout.tie(nullptr);

    int n; cin >> n;
    for (int i = 0; i < n; i++)
    {
        int m; cin >> m;
        int x, y; cin >> x >> y;
        Point start{ x, y }, pre{ x, y };
        for (int j = 1; j < m; j++)
        {
            cin >> x >> y;
            Point cur{ x, y };
            L[cnt++] = Line(pre, cur - pre);
            pre = cur;
        }
        L[cnt++] = Line(pre, start - pre);
    }

    HPI();
    double res = 0;
    for (int i = hh + 3, k = hh + 1; i <= tt; i++)
    {
        res += area(qp[k], qp[i - 1], qp[i]) / 2;
    }
    printf("%.3lf\n", res);

    return 0;
}
```



## 最小圆覆盖

### 最小圆覆盖

题目：https://www.acwing.com/problem/content/3031/

题解：《算法竞赛》

```c++
#include <cstdio>
#include <cmath>
#include <iostream>
#include <algorithm>
#include <numeric>

using LL = long long;
using std::cin, std::cout;

const int N = int(1e5) + 5;
const double EPS = 1e-12, PI = acos(-1);

int n;
struct Point
{
    double x, y;

    Point operator+ (const Point& o)
    {
        return { x + o.x, y + o.y };
    }
    Point operator- (const Point& o)
    {
        return { x - o.x, y - o.y };
    }
    Point operator* (double k)
    {
        return { x * k, y * k };
    }
    Point operator/ (double k)
    {
        return { x / k, y / k };
    }
} P[N];

struct Circle
{
    Point c;
    double r;
};

struct Line
{
    Point p, v;
};

int sign(double v)
{
    if (fabs(v) < EPS) return 0;
    return v < 0 ? -1 : 1;
}

double getDist(Point& a, Point& b)
{
    double dx = a.x - b.x, dy = a.y - b.y;
    return sqrt(dx * dx + dy * dy);
}

double cross(Point& a, Point& b)
{
    return a.x * b.y - a.y * b.x;
}

Point getIntersection(Line& a, Line& b)
{
    Point u = a.p - b.p;
    double t = cross(b.v, u) / cross(a.v, b.v);
    return a.p + (a.v * t);
}

bool inCircle(Circle& c, Point& p)
{
    return sign(c.r - getDist(c.c, p)) >= 0;
}

Point rotate(Point p, double k)  // 逆时针旋转
{
    return { p.x * cos(k) - p.y * sin(k), p.x * sin(k) + p.y * cos(k) };
}

Circle getCircle(Point& a, Point& b, Point& c)
{
    Circle res;
    Line u = { (a + b) / 2, rotate(b - a, PI / 2) };  // 求中垂线
    Line v = { (a + c) / 2, rotate(c - a ,PI / 2) };
    res.c = getIntersection(u, v);
    res.r = getDist(res.c, a);
    return res;
}

Circle mcc()
{
    std::random_shuffle(P, P + n);

    Circle c = { P[0], 0 };
    for (int i = 1; i < n; i++)
        if (!inCircle(c, P[i]))
        {
            c = { P[i], 0 };
            for (int j = 0; j < i; j++)
                if (!inCircle(c, P[j]))
                {
                    c = { (P[i] + P[j]) / 2, getDist(P[i], P[j]) / 2 };
                    for (int k = 0; k < j; k++)
                        if (!inCircle(c, P[k]))
                        {
                            c = getCircle(P[i], P[j], P[k]);
                        }
                }
        }

    return c;
}

int main()
{
    std::ios::sync_with_stdio(false);
    cin.tie(nullptr), cout.tie(nullptr);

    cin >> n;
    for (int i = 0; i < n; i++)
    {
        double x, y; cin >> x >> y;
        P[i] = { x, y };
    }
    Circle c = mcc();
    printf("%.10lf\n", c.r);
    printf("%.10lf %.10lf\n", c.c.x, c.c.y);

    return 0;
}
```

## 三维凸包

### 最佳包裹

题目：https://www.acwing.com/problem/content/2121/

题解：https://www.acwing.com/solution/content/169547/

```c++
#include <cstdio>
#include <cstdlib>
#include <ctime>
#include <cmath>
#include <iostream>
#include <algorithm>
#include <numeric>

using LL = long long;
using std::cin, std::cout;

const int N = 310;  // 欧拉多面体公式
const double EPS = 1e-10;

int n, m;

struct Point
{
    double x, y, z;

    static double randEPS()
    {
        return (double(rand()) / RAND_MAX - 0.5) * EPS;
    }

    void shake()
    {
        x += randEPS(), y += randEPS(), z += randEPS();
    }

    Point operator+ (const Point& o) const
    {
        return { x + o.x, y + o.y, z + o.z };
    }
    Point operator- (const Point & o) const
    {
        return { x - o.x, y - o.y, z - o.z };
    }
    Point operator* (double k) const
    {
        return { x * k, y * k, z * k };
    }
    Point operator/ (double k) const
    {
        return { x / k, y / k, z / k };
    }

    double len()
    {
        return sqrt(x * x + y * y + z * z);
    }
} P[N];

Point cross(const Point& a, const Point& b)
{
    return { a.y * b.z - b.y * a.z,
        b.x * a.z - a.x * b.z,
        a.x * b.y - a.y * b.x };
}

double dot(const Point& a, const Point& b)
{
    return a.x * b.x + a.y * b.y + a.z * b.z;
}

struct Plane
{
    int v[3];

    Point norm() const
    {
        return cross(P[v[1]] - P[v[0]], P[v[2]] - P[v[0]]);
    }

    double area() const
    {
        return norm().len() / 2;
    }
} cur[N], nxt[N];

bool abovePlane(const Plane& plane, const Point& point)
{
    return dot(plane.norm(), point - P[plane.v[0]]) >= 0;
}

void getConvexHull3D()
{
    static bool G[N][N];

    cur[m++] = { 0, 1, 2 };
    cur[m++] = { 2, 1, 0 };

    for (int i = 3; i < n; i++)
    {
        int cnt = 0;
        for (int j = 0; j < m; j++)
        {
            bool flag = abovePlane(cur[j], P[i]);
            if (!flag) nxt[cnt++] = cur[j];
            for (int k = 0; k < 3; k++)
            {
                int a = cur[j].v[k], b = cur[j].v[(k + 1) % 3];
                G[a][b] = flag;
            }
        }

        for (int j = 0; j < m; j++)
        {
            for (int k = 0; k < 3; k++)
            {
                int a = cur[j].v[k], b = cur[j].v[(k + 1) % 3];
                if (G[a][b] && !G[b][a])
                {
                    nxt[cnt++] = { a, b, i };
                }
            }
        }

        std::copy(nxt, nxt + cnt, cur);
        m = cnt;
    }
}

int main()
{

    cin.tie(nullptr), cout.tie(nullptr);

    for (int i = 0; i < n; i++)
    {
        double x, y, z; cin >> x >> y >> z;
        P[i] = { x, y, z };
        P[i].shake();
    }

    getConvexHull3D();

    double res = 0;
    for (int i = 0; i < m; i++)
    {
        res += cur[i].area();
```

## 旋转卡壳

### 周游世界

题目：https://www.acwing.com/problem/content/description/2941/

题解：https://www.acwing.com/solution/content/169935/

有两点要注意，

1. 如果所有的点都在一条直线上，那么需要特殊考虑
2. `rotatingCalipers`中j的范围是$[1, j]$

```c++
#include <cstdio>
#include <iostream>
#include <algorithm>
#include <numeric>

using LL = long long;
using std::cin, std::cout;

const int N = int(5e4) + 5;

int n;

struct Point
{
    int x, y;

    bool operator< (const Point& o) const
    {
        return x != o.x ? x < o.x : y < o.y;
    }

    Point operator+ (const Point& o) const
    {
        return { x + o.x, y + o.y };
    }
    Point operator- (const Point& o) const
    {
        return { x - o.x, y - o.y };
    }
    Point operator* (int k) const
    {
        return { x * k, y * k };
    }
} P[N];

int cross(const Point& a, const Point& b)
{
    return a.x * b.y - a.y * b.x;
}

int area(const Point& a, const Point& b, const Point& c)
{
    return cross(b - a, c - a);
}

int getDistSquare(const Point& a, const Point& b)
{
    int dx = a.x - b.x, dy = a.y - b.y;
    return dx * dx + dy * dy;
}

int stk[N], top;

void getConvexHull()  // Andrew算法
{
    static bool st[N];
    std::sort(P, P + n);
    //n = std::unique(P, P + n);  // 根据题意，不需要这一行

    for (int i = 0; i < n; i++)
    {
        while (top >= 2 && 
            area(P[stk[top - 1]], P[stk[top]], P[i]) >= 0)
        {
            st[stk[top--]] = false;
        }
        stk[++top] = i;
        st[i] = true;
    }
    st[0] = false;
    for (int i = n - 1; i >= 0; i--)
        if (!st[i])
        {
            while (top >= 2 &&
                area(P[stk[top - 1]], P[stk[top]], P[i]) >= 0)
            {
                top--;
            }
            stk[++top] = i;
        }
    top--;  // 防止0号点重复添加
}

int rotatingCalipers()
{
    if (top <= 2) return getDistSquare(P[0], P[n - 1]);

    int res = 0;
    for (int i = 1, j = 3; i <= top; i++)
    {
        const Point& a = P[stk[i]], & b = P[stk[i + 1]];
        while (area(P[stk[j]], b, a) < area(P[stk[j + 1]], b, a))
        {
            j = j % top + 1;  // 使得j的范围在[1, j]之间
        }
        res = std::max({ res, 
            getDistSquare(P[stk[j]], a), getDistSquare(P[stk[j]], b) });
    }
    return res;
}

int main()
{
    std::ios::sync_with_stdio(false);
    cin.tie(nullptr), cout.tie(nullptr);

    cin >> n;
    for (int i = 0; i < n; i++)
    {
        int x, y; cin >> x >> y;
        P[i] = { x, y };
    }

    getConvexHull();
    cout << rotatingCalipers() << '\n';
    return 0;
}
```

### 最小矩形覆盖

题目：https://www.acwing.com/problem/content/description/2144/

题解：https://www.acwing.com/solution/content/169970/

```c++
#include <cstdio>
#include <cmath>
#include <iostream>
#include <algorithm>
#include <numeric>

using LL = long long;
using std::cin, std::cout;

const int N = int(5e4) + 5;
const double INF = 1e20, EPS = 1e-12, PI = acos(-1);

int n;

struct Point
{
    double x, y;

    bool operator< (const Point& o) const
    {
        return x != o.x ? x < o.x : y < o.y;
    }

    Point operator+ (const Point& o) const
    {
        return { x + o.x, y + o.y };
    }
    Point operator- (const Point& o) const
    {
        return { x - o.x, y - o.y };
    }
    Point operator* (double k) const
    {
        return { x * k, y * k };
    }
    Point operator/ (double k) const
    {
        return { x / k, y / k };
    }

    double len() const
    {
        return sqrt(x * x + y * y);
    }
    Point unit() const
    {
        return *this / len();
    }
} P[N];

int sign(double v)
{
    if (fabs(v) < EPS) return 0;
    return v < 0 ? -1 : 1;
}

double cross(const Point& a, const Point& b)
{
    return a.x * b.y - a.y * b.x;
}

double dot(const Point& a, const Point& b)
{
    return a.x * b.x + a.y * b.y;
}

double area(const Point& a, const Point& b, const Point& c)
{
    return cross(b - a, c - a);
}

int stk[N], top;
bool st[N];

void getConvexHull()
{
    std::sort(P, P + n);
    for (int i = 0; i < n; i++)
    {
        while (top >= 2 && 
            area(P[stk[top - 1]], P[stk[top]], P[i]) <= 0)
        {
            st[stk[top--]] = false;
        }
        stk[++top] = i;
        st[i] = true;
    }

    st[0] = false;
    for (int i = n - 1; i >= 0; i--)
        if (!st[i])
        {
            while (top >= 2 &&
                area(P[stk[top - 1]], P[stk[top]], P[i]) <= 0)
            {
                top--;
            }
            stk[++top] = i;
        }
    top--;
}

Point res[4];  // 逆时针排序的4个点
double minArea = INF;

double project(const Point& a, const Point& b)  // 计算a在b上的投影 
{
    return dot(a, b) / b.len();
}

Point rotate(const Point& a, double r)
{
    return { a.x * cos(r) - a.y * sin(r), a.x * sin(r) + a.y * cos(r) };
}

void rotatingCalipers()
{
    for (int i = 1, a = 3, b = 2, c = 3; i <= top; i++)
    {
        const Point& d = P[stk[i]], & e = P[stk[i + 1]];
        while (sign(area(d, e, P[stk[a]]) - area(d, e, P[stk[a + 1]])) < 0)
        {
            a = a % top + 1;
        }
        while (sign(dot(e - d, P[stk[b]] - d) - dot(e - d, P[stk[b + 1]] - d)) < 0)
        {
            b = b % top + 1;
        }
        if (i == 1) c = a;
        while (sign(dot(d - e, P[stk[c]] - d) - dot(d - e, P[stk[c + 1]] - d)) < 0)
        {
            c = c % top + 1;
        }

        double w = project(P[stk[b]] - P[stk[c]], e - d);
        double h = area(d, e, P[stk[a]]) / (e - d).len();
        if (w * h < minArea)
        {
            minArea = w * h;
            res[0] = d + (e - d).unit() * project(P[stk[b]] - d, e - d);
            res[3] = d + (d - e).unit() * project(P[stk[c]] - d, d - e);
            Point up = rotate(e - d, PI / 2).unit();
            res[1] = res[0] + up * h;
            res[2] = res[3] + up * h;
        }
    }
}

int main()
{
    std::ios::sync_with_stdio(false);
    cin.tie(nullptr), cout.tie(nullptr);

    cin >> n;
    for (int i = 0; i < n; i++)
    {
        double x, y; cin >> x >> y;
        P[i] = { x , y };
    }

    getConvexHull();
    rotatingCalipers();

    int k = 0;
    for (int i = 1; i < 4; i++)
        if (sign(res[i].y - res[k].y) < 0 ||
            (!sign(res[i].y - res[k].y) && sign(res[i].x - res[k].x) < 0))
        {
            k = i;
        }
    printf("%.5lf\n", minArea);
    for (int i = 0; i < 4; i++)
    {
        if (!sign(res[k].x)) res[k].x = 0;
        if (!sign(res[k].y)) res[k].y = 0;
        printf("%.5lf %.5lf\n", res[k].x, res[k].y);
        k = (k + 1) % 4;
    }
    return 0;
}
```

## 三角剖分

### 望远镜

题目：https://www.acwing.com/problem/content/description/3037/

题解：https://www.acwing.com/solution/content/61667/

```c++
#include <cstdio>
#include <cmath>
#include <iostream>
#include <algorithm>
#include <numeric>

using LL = long long;
using std::cin, std::cout;

const int N = 55;
const double EPS = 1e-8, PI = acos(-1);

double R;
int n;

struct Point
{
    double x, y;

    Point operator+ (const Point& o) const
    {
        return { x + o.x, y + o.y };
    }
    Point operator- (const Point& o) const
    {
        return { x - o.x, y - o.y };
    }
    Point operator* (double k) const
    {
        return { x * k, y * k };
    }
    Point operator/ (double k) const
    {
        return { x / k, y / k };
    }

    double len() const
    {
        return sqrt(x * x + y * y);
    }

    Point unit() const
    {
        return *this / len();
    }
} P[N];

double cross(const Point& a, const Point& b)
{
    return a.x * b.y - a.y * b.x;
}

double dot(const Point& a, const Point& b)
{
    return a.x * b.x + a.y * b.y;
}

struct LineSeg
{
    Point a, b;
};

struct Line
{
    Point p, v;
};

int sign(double x)
{
    if (fabs(x) < EPS) return 0;
    return x < 0 ? -1 : 1;
}

double getDist(const Point& a, const Point& b)
{
    double dx = a.x - b.x, dy = a.y - b.y;
    return sqrt(dx * dx + dy * dy);
}

Point O = { 0, 0 };

double area(const Point& a, const Point& b, const Point& c)
{
    return cross(b - a, c - a);
}

Point getIntersectionPoint(const Line& a, const Line& b)
{
    Point u = a.p - b.p;
    double t = cross(b.v, u) / cross(a.v, b.v);
    return a.p + a.v * t;
}

Point rotate(const Point& p, double x)
{
    return { p.x * cos(x) - p.y * sin(x), p.x * sin(x) + p.y * cos(x) };
}

bool onLineSeg(const LineSeg& ls, const Point& p)
{
    Point ap = p - ls.a, bp = p - ls.b;
    return !sign(cross(ap, bp)) && sign(dot(ap, bp)) <= 0;
}

double getLineSegCircleDist(const LineSeg& ls, Point& pa, Point& pb)
{
    Point u = rotate(ls.b - ls.a, PI / 2);
    Point foot = getIntersectionPoint({ O, u - O }, { ls.a, ls.b - ls.a });
    double res, dist = getDist(foot, O);
    if (!onLineSeg(ls, foot))
    {
        res = std::min(getDist(ls.a, O), getDist(ls.b, O));
    }
    else res = dist;
    if (sign(res - R) >= 0) return res;

    double len = sqrt(R * R - dist * dist);
    pa = foot + (ls.a - ls.b).unit() * len;
    pb = foot + (ls.b - ls.a).unit() * len;
    return res;
}

double getSectorArea(const Point& a, const Point& b)
{
    double t = acos(dot(a, b) / (a.len() * b.len()));
    if (sign(area(O, a, b)) < 0) t = -t;
    return R * R * (t / 2);
}

double getCircleTriangleArea(const Point& a, const Point& b)
{
    double da = getDist(a, O), db = getDist(b, O);
    if (sign(da - R) <= 0 && sign(db - R) <= 0)
    {
        return area(O, a, b) / 2;
    }
    if (!sign(area(O, a, b))) return 0;

    Point pa, pb;
    double minDist = getLineSegCircleDist({ a, b }, pa, pb);
    if (sign(minDist - R) >= 0) return getSectorArea(a, b);
    if (sign(da - R) <= 0)
    {
        return getSectorArea(pb, b) + area(O, a, pb) / 2;
    }
    if (sign(db - R) <= 0)
    {
        return getSectorArea(a, pa) + area(O, pa, b) / 2;
    }
    return getSectorArea(a, pa) + area(O, pa, pb) / 2 + getSectorArea(pb, b);
}

void solve()
{
    for (int i = 0; i < n; i++)
    {
        double x, y; cin >> x >> y;
        P[i] = { x, y };
    }

    double sum = 0;
    for (int i = 0; i < n; i++)
    {
        sum += getCircleTriangleArea(P[i], P[(i + 1) % n]);
    }
    printf("%.2lf\n", fabs(sum));
}

int main()
{
    std::ios::sync_with_stdio(false);
    cin.tie(nullptr), cout.tie(nullptr);

    while (cin >> R >> n)
        solve();

    return 0;
}
```

### 扫描线

题目：https://www.acwing.com/problem/content/3071/

```c++
#include <cstdio>
#include <iostream>
#include <algorithm>
#include <numeric>
#include <vector>

using LL = long long;

{

    {
        if (x != o.x) return x < o.x;
        return y < o.y;
    }

using Range = Point;
Range R[N];

LL rangeArea(int l, int r)
{
    int cnt = 0;
    for (int i = 0; i < n; i++)
        if (ld[i].x <= l && ru[i].x >= r)
        {
            R[cnt++] = { ld[i].y, ru[i].y };
        }
    if (!cnt) return 0;

    std::sort(R, R + cnt);
    LL res = 0;
    auto [pre, cur] = R[0];
    for (int i = 1; i < cnt; i++)
    {
        if (R[i].x > cur)
        {
            res += cur - pre;
            pre = R[i].x, cur = R[i].y;
        }
        else cur = std::max(cur, R[i].y);
    }
    res += cur - pre;
    return res * (r - l);
}

{

        disc.push_back(ld[i].x);
        disc.push_back(ru[i].x);
    }
    std::sort(disc.begin(), disc.end());
    disc.erase(std::unique(disc.begin(), disc.end()), disc.end());

    LL res = 0;
    for (int i = 0; i + 1 < disc.size(); i++)
    {
        res += rangeArea(disc[i], disc[i + 1]);
```

### 三角形面积并

题目：https://www.acwing.com/problem/content/2803/

题解：https://www.acwing.com/solution/content/170683/

```c++
#include <cstdio>
#include <iostream>
#include <algorithm>
#include <numeric>
#include <vector>

using LL = long long;
using std::cin, std::cout;

const int N = 110;
const double EPS = 1e-8, INF = 1e7;

struct Point
{
    double x, y;

    bool operator< (const Point& o) const
    {
        return x != o.x ? x < o.x : y < o.y;
    }

    {
        return { x + o.x, y + o.y };
    }
    Point operator- (const Point& o) const
    {
        return { x - o.x, y - o.y };
    }
    Point operator* (double k) const
    {
        return { x * k, y * k };
    }
    Point operator/ (double k) const
    {
        return { x / k, y / k };
    }
};

double cross(const Point& a, const Point& b)
{
    return a.x * b.y - a.y * b.x;
}

double dot(const Point& a, const Point& b)
{
    return a.x * b.x + a.y * b.y;
}

int sign(double v)
{
    if (std::abs(v) < EPS) return 0;
    return v < 0 ? -1 : 1;
}

int cmp(double a, double b)
{
    return sign(a - b);
}

int n;
struct Triangle
{
    Point p[3];
} tr[N];

std::vector<double> xs;

struct Line
{
    Point p, v;
};

Point getIntersectionPoint(const Line& a, const Line& b)
{
    Point u = a.p - b.p;
    double t = cross(b.v, u) / cross(a.v, b.v);
    return a.p + a.v * t;
}

// p已经在(a, b)构成的直线上，判断p是否在(a, b)构成的线段上
bool onLineSeg(const Point& a, const Point& b, const Point& p)  
{
    return sign(dot(p - a, p - b)) <= 0;
}

double getIntersectionLength(double x, int side)
{
    using Range = Point;
    static Range R[N];  // 区间合并
    int cnt = 0;

    for (int i = 0; i < n; i++)
    {
        const auto& t = tr[i];
        if (cmp(t.p[0].x, x) > 0 || cmp(t.p[2].x, x) < 0) continue;
        if (side && !cmp(t.p[0].x, x) && !cmp(t.p[1].x, x))
        {
            R[cnt++] = { std::min(t.p[0].y, t.p[1].y),
                         std::max(t.p[0].y, t.p[1].y) };
        }
        else if (!side && !cmp(t.p[1].x, x) && !cmp(t.p[2].x, x))
        {
            R[cnt++] = { std::min(t.p[1].y, t.p[2].y),
                         std::max(t.p[1].y, t.p[2].y) };
        }
        else
        {
            std::vector<double> ys;
            for (int j = 0; j < 3; j++)
            {
                Point p = getIntersectionPoint(
                    { t.p[j], t.p[(j + 1) % 3] - t.p[j] }, 
                    { {x, 0}, {0, 1} });
                if (onLineSeg(t.p[j], t.p[(j + 1) % 3], p))
                {
                    ys.push_back(p.y);
                }
            }
            if (ys.size() >= 2)
            {
                std::sort(ys.begin(), ys.end());
                R[cnt++] = { ys.front(), ys.back() };
            }
        }
    }

    if (!cnt) return 0;
    std::sort(R, R + cnt);
    double res = 0;
    auto [st, ed] = R[0];
    for (int i = 1; i < cnt; i++)
    {
        if (R[i].x < ed) ed = std::max(ed, R[i].y);
        else
        {
            res += ed - st;
            st = R[i].x, ed = R[i].y;
        }
    }
    res += ed - st;
    return res;
}

double rangeArea(double a, double b)
{
    return 
        (getIntersectionLength(a, 1) + 
         getIntersectionLength(b, 0))
        * (b - a) / 2;
}

int main()
{
    std::ios::sync_with_stdio(false);
    cin.tie(nullptr), cout.tie(nullptr);

    cin >> n;
    for (int i = 0; i < n; i++)
    {
        for (int j = 0; j < 3; j++)
        {
            double x, y; cin >> x >> y;
            tr[i].p[j] = { x, y };
            xs.push_back(x);
        }
        std::sort(tr[i].p, tr[i].p + 3);
    }

    for (int i = 0; i < n; i++)
        for (int j = 0; j < i; j++)
            for (int x = 0; x < 3; x++)
                for (int y = 0; y < 3; y++)
                {
                    Point p = getIntersectionPoint(
                        { tr[i].p[x], tr[i].p[(x + 1) % 3] - tr[i].p[x] },
                        { tr[j].p[y], tr[j].p[(y + 1) % 3] - tr[j].p[y] });
                    if (onLineSeg(tr[i].p[x], tr[i].p[(x + 1) % 3], p) &&
                        onLineSeg(tr[j].p[y], tr[j].p[(y + 1) % 3], p))
                    {
                        xs.push_back(p.x);
                    }
                }

    double res = 0;
    std::sort(xs.begin(), xs.end());
    for (int i = 0; i + 1 < xs.size(); i++)
        if (cmp(xs[i], xs[i + 1]))
        {
            res += rangeArea(xs[i], xs[i + 1]);
        }
    printf("%.2lf\n", res);

    return 0;
}
```
